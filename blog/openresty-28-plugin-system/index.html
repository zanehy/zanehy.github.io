28 | 插件系统设计与扩展

发表于： 2026-04-03 分类于： OpenResty
字数： 1800 阅读：≈ 4分钟 浏览：

前几篇写了路由、鉴权、限流、熔断、改写这些具体插件。这篇来把插件系统本身做完整——自动加载、优先级排序、生命周期钩子，以及如何让别人能轻松写新插件。

插件加载机制

目标：把 plugins/ 目录下的所有 Lua 文件自动识别为插件，不需要手动注册。

-- core/plugin_runner.lua
local M = {}

-- 已加载的插件列表
local _plugins = {}
local _loaded   = false

-- 扫描并加载所有插件
local function load_plugins()
    if _loaded then return end

    -- 内置插件列表（按文件名，实际项目可以用 lfs 扫描目录）
    local plugin_names = {
        "canary",           -- priority: 5
        "ip_blacklist",     -- priority: 10
        "rate_limit",       -- priority: 12
        "jwt_auth",         -- priority: 20
        "apikey_auth",      -- priority: 21
        "circuit_breaker",  -- priority: 25 (access + log)
        "request_rewrite",  -- priority: 30
        "logging",          -- priority: 90 (log only)
    }

    for _, name in ipairs(plugin_names) do
        local ok, plugin = pcall(require, "plugins." .. name)
        if ok and plugin then
            _plugins[#_plugins + 1] = plugin
            ngx.log(ngx.DEBUG, "plugin loaded: ", name, " priority=", plugin.priority or 100)
        else
            ngx.log(ngx.ERR, "failed to load plugin: ", name, " err=", tostring(plugin))
        end
    end

    -- 按优先级排序
    table.sort(_plugins, function(a, b)
        return (a.priority or 100) < (b.priority or 100)
    end)

    _loaded = true
    ngx.log(ngx.INFO, "loaded ", #_plugins, " plugins")
end

-- 获取当前路由的插件配置
local function get_plugin_config(plugin_name, route)
    -- 路由级配置优先
    if route and route.plugins and route.plugins[plugin_name] then
        return route.plugins[plugin_name]
    end
    -- 全局配置兜底
    local global = ngx.shared.routes:get("global_config")
    if global then
        local cjson = require "cjson.safe"
        local cfg = cjson.decode(global)
        if cfg and cfg.plugins and cfg.plugins[plugin_name] then
            return cfg.plugins[plugin_name]
        end
    end
    return nil
end

-- 执行 access 阶段所有插件
function M.run_access()
    load_plugins()

    local route = ngx.ctx.route

    for _, plugin in ipairs(_plugins) do
        if not plugin.access then goto continue end

        local config = get_plugin_config(plugin.name, route)
        if not config or config.enabled == false then goto continue end

        local ok, err = pcall(plugin.access, plugin, config)
        if not ok then
            ngx.log(ngx.ERR, "plugin [", plugin.name, "] access error: ", err)
        end

        -- 如果请求已被终止（ngx.exit 被调用），停止执行后续插件
        -- 注意：ngx.exit 会抛出特殊错误，pcall 不会捕获到 ngx.exit
        ::continue::
    end
end

-- 执行 header_filter 阶段插件
function M.run_header_filter()
    load_plugins()
    local route = ngx.ctx.route

    for _, plugin in ipairs(_plugins) do
        if not plugin.header_filter then goto continue end
        local config = get_plugin_config(plugin.name, route)
        if not config or config.enabled == false then goto continue end
        pcall(plugin.header_filter, plugin, config)
        ::continue::
    end
end

-- 执行 log 阶段插件
function M.run_log()
    load_plugins()
    local route = ngx.ctx.route

    for _, plugin in ipairs(_plugins) do
        if not plugin.log then goto continue end
        local config = get_plugin_config(plugin.name, route)
        if not config or config.enabled == false then goto continue end
        pcall(plugin.log, plugin, config)
        ::continue::
    end
end

return M

插件执行顺序与优先级

优先级数字越小，越早执行。设计参考：

优先级范围适合的插件类型
1 ~ 9流量染色（灰度、AB测试）
10 ~ 19安全防护（IP黑名单、WAF）
20 ~ 29鉴权（JWT、API Key）
30 ~ 49限流、熔断
50 ~ 79请求改写
80 ~ 100日志、监控

鉴权必须在限流之前——已经被限流拒绝的请求，没必要再做鉴权计算，这样能节省资源。而流量染色（决定路由到哪个 upstream）要最先执行，后续插件拿到的才是最终的路由信息。

插件生命周期钩子

完整的插件接口，支持多个阶段：

-- 插件接口模板
local MyPlugin = {}
MyPlugin.name     = "my_plugin"
MyPlugin.priority = 50

-- access 阶段：处理请求，可以终止请求（调用 ngx.exit）
function MyPlugin:access(config)
end

-- header_filter 阶段：修改响应头（不能读 body，不能调用 cosocket）
function MyPlugin:header_filter(config)
end

-- body_filter 阶段：修改响应体
function MyPlugin:body_filter(config)
end

-- log 阶段：记录日志、更新统计（不能修改响应）
function MyPlugin:log(config)
end

return MyPlugin

并不是每个插件都需要实现所有钩子，只实现需要的即可。插件执行器会检查方法是否存在再调用。

如何写一个自定义插件

以一个"请求签名验证"插件为例，演示完整的开发流程：

-- plugins/request_sign.lua
-- 功能：验证请求是否携带有效的签名（适合服务间调用）

local hmac   = require "resty.hmac"
local Plugin = {}

Plugin.name     = "request_sign"
Plugin.priority = 22  -- 在 JWT 之后，限流之前

function Plugin:access(config)
    local headers = ngx.req.get_headers()

    -- 读取签名相关头
    local app_id    = headers["x-app-id"]
    local timestamp = headers["x-timestamp"]
    local signature = headers["x-signature"]

    if not app_id or not timestamp or not signature then
        return require("lib.response").error(401, "missing signature headers", 401)
    end

    -- 防重放：时间戳不能超过 5 分钟
    local ts = tonumber(timestamp)
    if not ts or math.abs(ngx.time() - ts) > 300 then
        return require("lib.response").error(401, "timestamp expired", 401)
    end

    -- 从配置或 Redis 获取 app_secret
    local secret = config.secrets and config.secrets[app_id]
    if not secret then
        return require("lib.response").error(401, "unknown app_id", 401)
    end

    -- 重新计算签名：HMAC-SHA256(method + uri + timestamp, secret)
    local method = ngx.req.get_method()
    local uri    = ngx.var.uri
    local sign_str = method .. uri .. timestamp

    local h = hmac:new(secret, hmac.ALGOS.SHA256)
    h:update(sign_str)
    local expected = h:final(nil, true)  -- hex 编码

    if signature ~= expected then
        ngx.log(ngx.WARN, "invalid signature for app=", app_id, " uri=", uri)
        return require("lib.response").error(401, "invalid signature", 401)
    end

    ngx.ctx.app_id = app_id
end

return Plugin

把这个文件放到 plugins/ 目录，在 plugin_runner.lua 的 plugin_names 列表里加上 "request_sign"，插件就生效了。

插件配置的动态更新

插件配置存在路由配置里（Redis），每 30 秒自动刷新。想立即生效，调用管理接口：

location /admin/reload-config {
    allow 10.0.0.0/8;
    deny  all;
    content_by_lua_block {
        require("core.config_loader").reload()
        ngx.say('{"code":0}')
    }
}

更新某个路由的限流配置，不需要重启 nginx：

# 通过管理 API 更新 Redis 里的路由配置
curl -X PUT http://gateway-admin/routes/user-service \
  -H "Content-Type: application/json" \
  -d '{"plugins":{"rate_limit":{"enabled":true,"rpm":2000}}}'

# 立即生效
curl http://gateway/admin/reload-config

插件测试

每个插件都应该有独立的单元测试，用 busted 框架：

-- spec/plugins/rate_limit_spec.lua
local busted = require "busted"
local describe, it, assert = describe, it, assert

-- mock ngx
_G.ngx = {
    shared = { rate_limit = require "spec.mock.shared_dict" },
    var    = { remote_addr = "127.0.0.1" },
    log    = function() end,
    exit   = function(code) error("exit:" .. code) end,
    OK     = 200,
    time   = function() return 1000000 end,
}

local rate_limit = require "plugins.rate_limit"

describe("rate_limit plugin", function()
    it("allows requests within limit", function()
        local config = { rpm = 10, key = "ip" }
        -- 前 10 次应该通过
        for i = 1, 10 do
            assert.has_no_error(function()
                rate_limit:access(config)
            end)
        end
    end)

    it("rejects requests over limit", function()
        local config = { rpm = 10, key = "ip" }
        -- 第 11 次应该被限流
        assert.has_error(function()
            rate_limit:access(config)
        end, "exit:429")
    end)
end)

小结

一个健壮的插件系统需要：

  • 自动加载：按约定放文件，系统自动识别，无需手动注册
  • 优先级控制：用数字明确执行顺序，鉴权在限流前，限流在改写前
  • 多阶段钩子：access / header_filter / body_filter / log，插件按需实现
  • 动态配置：配置存 Redis，热更新无需重启，管理接口立即触发
  • 可测试：每个插件独立，可以 mock ngx 环境做单元测试
