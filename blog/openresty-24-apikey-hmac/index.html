24 | 鉴权模块：API Key + HMAC 签名

发表于： 2026-03-31 分类于： OpenResty
字数： 1941 阅读：≈ 4分钟 浏览：

JWT 适合用户身份认证，但服务间调用（机器对机器）通常用 API Key + HMAC 签名。这种方式不需要登录流程，调用方持有 API Key 和 Secret，每次请求用 Secret 对请求内容签名，网关验证签名的合法性。

API Key 设计规范

一对 API 凭证包含两部分：

  • API Key（app_key）：公开标识符，类似用户名，用于查找对应的 Secret。格式示例：ak_live_a1b2c3d4e5f6
  • API Secret（app_secret）：私密密钥，类似密码，只有调用方知道，用于生成签名。网关侧存储其哈希值。
前缀:   ak_         -- api key
环境:   live_ / test_   -- 区分生产和测试
随机串:  a1b2c3d4e5f6    -- 32位随机十六进制

在 Redis 里存储 API Key 信息：

# key: apikey:<app_key>
# value: JSON
redis-cli set "apikey:ak_live_a1b2c3d4e5f6" '{
  "app_key": "ak_live_a1b2c3d4e5f6",
  "app_secret_hash": "<bcrypt hash of secret>",
  "app_name": "order-service",
  "permissions": ["read", "write"],
  "rate_limit": 1000,
  "enabled": true
}'

注意：存储的是 Secret 的哈希值，不是明文。但 HMAC 验证需要原始 Secret，所以实际上有两种存储方案：

  1. 存明文 Secret：简单，但有泄露风险
  2. 存密文，签名时重新计算：更安全，但需要在签名协议中让调用方传 Secret 原文……这显然不合理

实践中，API Secret 通常存明文但加密保存（用 AES 加密存储，读取时解密），或者接受"内部系统信任边界"的前提直接存明文。下文示例用明文存储以保持清晰。

HMAC-SHA256 签名协议

签名协议决定了调用方和网关如何就"待签名内容"达成一致。一个典型的设计：

待签名字符串构造 

StringToSign = Method + "\n"
             + Path + "\n"
             + QueryString（按 key 字典序排列）+ "\n"
             + Timestamp + "\n"
             + Nonce + "\n"
             + MD5(RequestBody)  （GET 请求为空字符串）

请求示例

调用方发出的请求：

POST /api/v1/order HTTP/1.1
Host: gateway.example.com
X-App-Key: ak_live_a1b2c3d4e5f6
X-Timestamp: 1710000000
X-Nonce: a8f3b291
X-Signature: 3d4e5f6a7b8c9d0e1f2a3b4c5d6e7f80

{"product_id": 123, "quantity": 2}

签名计算（调用方）：

import hmac, hashlib, time, os

app_key    = "ak_live_a1b2c3d4e5f6"
app_secret = "your_secret"
timestamp  = str(int(time.time()))
nonce      = os.urandom(4).hex()
body       = '{"product_id": 123, "quantity": 2}'
body_md5   = hashlib.md5(body.encode()).hexdigest()

string_to_sign = "\n".join([
    "POST",
    "/api/v1/order",
    "",          # 无 query string
    timestamp,
    nonce,
    body_md5
])

signature = hmac.new(
    app_secret.encode(),
    string_to_sign.encode(),
    hashlib.sha256
).hexdigest()

实现 HMAC 鉴权插件 

-- plugins/apikey_auth.lua
local redis    = require "resty.redis"
local cjson    = require "cjson"
local response = require "lib.response"

local Plugin = {}
Plugin.name     = "apikey_auth"
Plugin.priority = 20

-- 从 Redis 获取 app 信息
local function get_app(app_key)
    local red = redis:new()
    red:set_timeouts(500, 500, 500)
    local ok = red:connect("127.0.0.1", 6379)
    if not ok then return nil, "redis unavailable" end

    local data = red:get("apikey:" .. app_key)
    red:set_keepalive(10000, 50)

    if not data or data == ngx.null then
        return nil, "app not found"
    end
    return cjson.decode(data)
end

-- 计算 HMAC-SHA256
local function hmac_sha256(secret, message)
    local resty_hmac = require "resty.hmac"
    local hmac = resty_hmac:new(secret, resty_hmac.ALGOS.SHA256)
    hmac:update(message)
    return hmac:final(nil, true)  -- true = hex 格式
end

-- 构造待签名字符串
local function build_string_to_sign(method, path, query, timestamp, nonce, body_md5)
    -- query string 按 key 字典序排列
    local sorted_query = ""
    if query and query ~= "" then
        local args = ngx.decode_args(query)
        local keys = {}
        for k in pairs(args) do keys[#keys + 1] = k end
        table.sort(keys)
        local parts = {}
        for _, k in ipairs(keys) do
            parts[#parts + 1] = k .. "=" .. (args[k] or "")
        end
        sorted_query = table.concat(parts, "&")
    end

    return table.concat({
        method,
        path,
        sorted_query,
        timestamp,
        nonce,
        body_md5,
    }, "\n")
end

function Plugin:access(config)
    local headers   = ngx.req.get_headers()
    local app_key   = headers["x-app-key"]
    local timestamp = headers["x-timestamp"]
    local nonce     = headers["x-nonce"]
    local signature = headers["x-signature"]

    -- 检查必要头部
    if not app_key or not timestamp or not nonce or not signature then
        return response.error(401, "missing auth headers", 401)
    end

    -- 防重放：检查时间戳偏差（允许 ±5 分钟）
    local ts = tonumber(timestamp)
    if not ts or math.abs(ngx.time() - ts) > 300 then
        return response.error(401, "timestamp expired or invalid", 401)
    end

    -- 防重放：检查 nonce 是否已使用
    local nonce_key = "nonce:" .. app_key .. ":" .. nonce
    local nonce_cache = ngx.shared.nonce_cache
    if nonce_cache then
        local exists = nonce_cache:get(nonce_key)
        if exists then
            return response.error(401, "nonce already used", 401)
        end
        -- 记录 nonce，TTL 设为时间窗口的 2 倍
        nonce_cache:set(nonce_key, 1, 600)
    end

    -- 从 Redis 获取 app 信息（含 secret）
    local app, err = get_app(app_key)
    if not app then
        ngx.log(ngx.WARN, "unknown app_key: ", app_key)
        return response.error(401, "invalid app_key", 401)
    end

    if not app.enabled then
        return response.error(403, "app disabled", 403)
    end

    -- 读取请求 body
    ngx.req.read_body()
    local body     = ngx.req.get_body_data() or ""
    local body_md5 = ngx.md5(body)

    -- 构造待签名字符串
    local string_to_sign = build_string_to_sign(
        ngx.req.get_method(),
        ngx.var.uri,
        ngx.var.query_string or "",
        timestamp,
        nonce,
        body_md5
    )

    -- 计算期望签名
    local expected_sig = hmac_sha256(app.app_secret, string_to_sign)

    -- 比较签名（防时序攻击：用等长比较）
    if #expected_sig ~= #signature or expected_sig ~= signature then
        ngx.log(ngx.WARN, "signature mismatch, app=", app_key)
        return response.error(401, "invalid signature", 401)
    end

    -- 鉴权通过，注入 app 信息
    ngx.ctx.app = {
        key         = app_key,
        name        = app.app_name,
        permissions = app.permissions,
    }
    ngx.req.set_header("X-App-Key",  app_key)
    ngx.req.set_header("X-App-Name", app.app_name)
end

return Plugin

防重放攻击详解

重放攻击是指攻击者截获一个合法请求后，直接重复发送这个请求。即使签名是正确的，也不应该被执行两次（比如重复下单、重复转账）。

双重防护：

  1. Timestamp 检查：请求时间戳与服务器时间差超过 5 分钟的请求直接拒绝。攻击者无法把一个请求"留着"以后用。

  2. Nonce 检查：每个请求携带一个随机字符串（nonce），服务器记录已经使用过的 nonce，在时间窗口内同一个 nonce 不能使用两次。

-- nonce 存储在共享内存里，配置 nonce_cache
-- nginx.conf 里声明：
-- lua_shared_dict nonce_cache 10m;

两者结合：timestamp 保证请求的时效性，nonce 保证同一请求不被重复使用。

实战：对接第三方系统

第三方系统（如支付回调、短信平台通知）通常有自己的签名规范，不完全符合上面的协议。这时需要为每个第三方写一个适配器。

以支付宝回调为例（简化版）：

-- plugins/alipay_notify_verify.lua
local Plugin = {}
Plugin.name     = "alipay_notify"
Plugin.priority = 20

function Plugin:access(config)
    ngx.req.read_body()
    local args = ngx.req.get_post_args()

    -- 支付宝签名规范：所有参数按字典序排列，sign 和 sign_type 除外
    local keys = {}
    for k in pairs(args) do
        if k ~= "sign" and k ~= "sign_type" then
            keys[#keys + 1] = k
        end
    end
    table.sort(keys)

    local parts = {}
    for _, k in ipairs(keys) do
        if args[k] and args[k] ~= "" then
            parts[#parts + 1] = k .. "=" .. args[k]
        end
    end

    local sign_str   = table.concat(parts, "&")
    local sign       = args.sign
    local sign_type  = args.sign_type or "RSA2"

    -- 用支付宝公钥验证签名（RSA2 = SHA256withRSA）
    -- 实际实现需要 lua-resty-rsa 或调用 openssl 命令
    local verify_ok = verify_rsa_signature(sign_str, sign, config.alipay_public_key)

    if not verify_ok then
        ngx.status = 200
        ngx.say("fail")   -- 支付宝要求返回 "fail" 表示验证失败
        return ngx.exit(ngx.OK)
    end

    -- 验证通过，继续处理回调
    ngx.ctx.notify_args = args
end

return Plugin

小结

API Key + HMAC 鉴权的核心要素：

  • API Key 用于标识调用方，Secret 用于签名，两者分开存储
  • 签名覆盖请求的关键内容：Method、Path、Body MD5 等，防止篡改
  • 防重放：时间戳检查（±5 分钟窗口）+ Nonce 检查（共享内存记录已用 nonce）
  • 对接第三方时按对方签名规范实现，不要强求统一
