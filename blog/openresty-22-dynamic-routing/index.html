22 | 路由模块：动态路由

发表于： 2026-03-30 分类于： OpenResty
字数： 1565 阅读：≈ 4分钟 浏览：

架构设计好了，这篇来实现第一个核心模块：路由。路由模块要解决两个问题——怎么匹配请求，怎么找到对应的后端服务。

路由匹配策略

路由匹配通常有三种方式，按精确度从高到低：

  1. 精确匹配：/api/v1/user/profile 只匹配这一个路径
  2. 前缀匹配：/api/v1/user 匹配所有以此开头的路径
  3. 正则匹配：/api/v[0-9]+/user 用正则表达式匹配

匹配优先级：精确 > 前缀（长的优先） > 正则。

-- core/router.lua
local cjson = require "cjson"
local M = {}

-- 路由表（从共享内存加载）
local function get_routes()
    local cache = ngx.shared.routes
    local data  = cache:get("routes")
    if not data then return {} end
    return cjson.decode(data) or {}
end

-- 路由匹配
function M.match(uri, method)
    local routes = get_routes()
    local matched = nil
    local matched_len = 0

    for _, route in ipairs(routes) do
        if not route.enabled then goto continue end

        -- 方法过滤
        if route.methods then
            local method_ok = false
            for _, m in ipairs(route.methods) do
                if m == method then method_ok = true; break end
            end
            if not method_ok then goto continue end
        end

        -- 精确匹配
        if route.path == uri then
            return route
        end

        -- 前缀匹配（取最长匹配）
        if route.path_prefix then
            local prefix = route.path_prefix
            if uri:sub(1, #prefix) == prefix then
                -- 确保匹配的是完整路径段（避免 /api/user 匹配 /api/username）
                local next_char = uri:sub(#prefix + 1, #prefix + 1)
                if next_char == "" or next_char == "/" or next_char == "?" then
                    if #prefix > matched_len then
                        matched     = route
                        matched_len = #prefix
                    end
                end
            end
        end

        -- 正则匹配
        if route.path_regex and not matched then
            local m = ngx.re.match(uri, route.path_regex, "jo")
            if m then
                matched = route
            end
        end

        ::continue::
    end

    return matched
end

-- 计算转发路径（处理 strip_prefix）
function M.get_upstream_path(route, uri)
    if not route.strip_prefix then
        return uri
    end

    local prefix = route.path_prefix or ""
    local stripped = uri:sub(#prefix + 1)
    if stripped == "" then stripped = "/" end
    -- 保留 query string
    local query = ngx.var.query_string
    if query and query ~= "" then
        stripped = stripped .. "?" .. query
    end
    return stripped
end

return M

从 Redis 加载路由配置

路由配置存在 Redis 里，启动时加载，定期刷新：

-- core/config_loader.lua
local redis  = require "resty.redis"
local cjson  = require "cjson"
local M = {}

local ROUTE_KEY = "gateway:routes"

local function get_redis()
    local red = redis:new()
    red:set_timeouts(2000, 2000, 2000)
    local ok, err = red:connect("127.0.0.1", 6379)
    if not ok then return nil, err end
    return red
end

function M.load()
    local red, err = get_redis()
    if not red then
        ngx.log(ngx.ERR, "config load failed, redis unavailable: ", err)
        return false
    end

    local data, err = red:get(ROUTE_KEY)
    red:set_keepalive(10000, 10)

    if not data or data == ngx.null then
        ngx.log(ngx.WARN, "no route config found in redis, using empty routes")
        ngx.shared.routes:set("routes", "[]")
        return true
    end

    -- 验证 JSON 合法性
    local routes, err = cjson.decode(data)
    if not routes then
        ngx.log(ngx.ERR, "invalid route config: ", err)
        return false
    end

    -- 写入共享内存
    local ok, err = ngx.shared.routes:set("routes", data)
    if not ok then
        ngx.log(ngx.ERR, "failed to cache routes: ", err)
        return false
    end

    ngx.log(ngx.INFO, "routes loaded: ", #routes, " rules")
    return true
end

-- 热更新：重新加载配置（不重启 nginx）
function M.reload()
    local ok = M.load()
    if ok then
        ngx.log(ngx.INFO, "routes reloaded at ", ngx.now())
    end
end

return M

在 Redis 里写入路由配置：

# 用 redis-cli 写入（生产环境通过管理 API 写入）
redis-cli set gateway:routes '[
  {
    "id": "user-service",
    "path_prefix": "/api/v1/user",
    "upstream": {"targets": [{"host": "127.0.0.1", "port": 8081}]},
    "strip_prefix": true,
    "enabled": true
  }
]'

热更新路由

路由变更后，不需要重启 nginx，只需要更新 Redis 里的配置，等下一次定时刷新（最多 30 秒），或者调用管理接口立即触发重载：

# 管理接口（只暴露在内网）
location /admin/reload-routes {
    allow 10.0.0.0/8;
    deny  all;

    content_by_lua_block {
        local loader = require "core.config_loader"
        local ok = loader.reload()
        if ok then
            ngx.say('{"code":0,"message":"routes reloaded"}')
        else
            ngx.status = 500
            ngx.say('{"code":500,"message":"reload failed"}')
        end
    }
}

upstream 动态选择

每条路由可以配置多个 upstream 目标，网关需要在它们之间做负载均衡。

-- core/balancer.lua
local balancer = require "ngx.balancer"
local M = {}

-- 轮询计数器（每个路由一个）
local round_robin_counters = {}

-- 从路由配置选取目标
local function select_target(route)
    local targets = route.upstream and route.upstream.targets
    if not targets or #targets == 0 then
        return nil, "no upstream targets"
    end

    if #targets == 1 then
        return targets[1]
    end

    local strategy = (route.upstream.strategy or "round_robin")

    if strategy == "round_robin" then
        local route_id = route.id or "default"
        local counter  = round_robin_counters[route_id] or 0
        counter = (counter % #targets) + 1
        round_robin_counters[route_id] = counter
        return targets[counter]

    elseif strategy == "random" then
        return targets[math.random(#targets)]

    elseif strategy == "ip_hash" then
        -- 按客户端 IP 哈希，同一 IP 始终路由到同一后端
        local ip  = ngx.var.remote_addr or "0.0.0.0"
        local idx = (tonumber(ngx.crc32_short(ip)) % #targets) + 1
        return targets[idx]
    end

    return targets[1]
end

-- balancer_by_lua 阶段调用
function M.balance()
    local route = ngx.ctx.route
    if not route then
        return balancer.set_current_peer("127.0.0.1", 80)
    end

    local target, err = select_target(route)
    if not target then
        ngx.log(ngx.ERR, "no target: ", err)
        return ngx.exit(502)
    end

    -- 设置超时
    balancer.set_timeouts(
        (route.upstream.connect_timeout or 2000) / 1000,
        (route.upstream.read_timeout    or 60000) / 1000,
        (route.upstream.send_timeout    or 60000) / 1000
    )

    local ok, err = balancer.set_current_peer(target.host, target.port)
    if not ok then
        ngx.log(ngx.ERR, "set_current_peer failed: ", err)
        return ngx.exit(502)
    end
end

return M

把路由结果传递给 balancer

balancer_by_lua 在 content_by_lua 之后执行，需要通过 ngx.ctx 传递路由信息：

-- content_by_lua_block（路由分发）
content_by_lua_block {
    local router   = require "core.router"
    local response = require "lib.response"

    local uri    = ngx.var.uri
    local method = ngx.req.get_method()

    local route, err = router.match(uri, method)
    if not route then
        return response.error(404, "no route found for " .. method .. " " .. uri, 404)
    end

    -- 存入 ctx，供 balancer_by_lua 使用
    ngx.ctx.route = route

    -- 计算转发路径
    local upstream_path = router.get_upstream_path(route, uri)
    ngx.req.set_uri(upstream_path)

    -- 设置 upstream Host
    local target = route.upstream.targets[1]
    ngx.var.upstream_host = target.host .. ":" .. target.port

    -- 转发到 @upstream location
    ngx.exec("@upstream")
}

路由调试接口

开发阶段方便测试路由是否匹配正确：

location /admin/route-test {
    allow 127.0.0.1;
    deny  all;

    content_by_lua_block {
        local router = require "core.router"
        local cjson  = require "cjson"

        local uri    = ngx.var.arg_uri or "/"
        local method = ngx.var.arg_method or "GET"

        local route = router.match(uri, method)
        if route then
            ngx.say(cjson.encode({matched = true, route_id = route.id, upstream = route.upstream}))
        else
            ngx.say(cjson.encode({matched = false}))
        end
    }
}

测试：

curl "http://gateway/admin/route-test?uri=/api/v1/user/123&method=GET"
# {"matched":true,"route_id":"user-service","upstream":{"targets":[...]}}

小结

路由模块的几个要点：

  • 匹配优先级：精确 > 前缀（最长优先） > 正则
  • 路由配置存 Redis，通过定时器热加载，管理接口可触发立即更新
  • ngx.ctx.route 是路由信息在各阶段之间传递的载体
  • 负载均衡策略在 balancer_by_lua 阶段执行，支持轮询、随机、IP Hash
