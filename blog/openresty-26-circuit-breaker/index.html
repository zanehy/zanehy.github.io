26 | 熔断与降级模块

发表于： 2026-04-02 分类于： OpenResty
字数： 1302 阅读：≈ 3分钟 浏览：

限流保护的是网关自身，熔断保护的是后端服务。当某个后端服务出现故障时，如果网关继续把请求转发过去，不仅请求会失败，还会占用连接资源，拖慢整个系统。熔断器的作用就是在检测到后端异常时，主动切断请求，给后端一个恢复的时间窗口。

熔断器三态模型

熔断器有三种状态，在这三种状态之间流转：

            失败率超过阈值
  关闭 ──────────────────→ 打开
  (正常转发)               (直接拒绝)
     ↑                        │
     │    探测请求成功          │ 超过重置时间
     └──────────────── 半开 ←──┘
                      (放行少量请求探测)
  • 关闭（Closed）：正常状态，请求正常转发。统计失败率，超过阈值则转为打开
  • 打开（Open）：熔断状态，所有请求直接返回降级响应，不转发到后端
  • 半开（Half-Open）：探测状态，打开一段时间后自动进入。放行少量请求探测后端是否恢复，成功则转为关闭，失败则重新转为打开

用 ngx.shared.DICT 实现熔断 

-- plugins/circuit_breaker.lua
local Plugin = {}
Plugin.name     = "circuit_breaker"
Plugin.priority = 15

-- 状态常量
local STATE_CLOSED    = "closed"
local STATE_OPEN      = "open"
local STATE_HALF_OPEN = "half_open"

local stats = ngx.shared.circuit_breaker

-- 获取熔断器状态
local function get_state(service_id)
    return stats:get("state:" .. service_id) or STATE_CLOSED
end

-- 记录请求结果
local function record_result(service_id, success)
    local window = 60  -- 统计窗口：60 秒

    if success then
        stats:incr("success:" .. service_id, 1, 0, window)
    else
        stats:incr("failure:" .. service_id, 1, 0, window)
    end
end

-- 计算失败率
local function failure_rate(service_id)
    local success = tonumber(stats:get("success:" .. service_id)) or 0
    local failure = tonumber(stats:get("failure:" .. service_id)) or 0
    local total   = success + failure
    if total == 0 then return 0 end
    return failure / total
end

-- 切换到打开状态
local function trip(service_id, reset_timeout)
    stats:set("state:" .. service_id, STATE_OPEN)
    stats:set("open_at:" .. service_id, ngx.time())
    stats:set("reset_timeout:" .. service_id, reset_timeout or 30)
    ngx.log(ngx.WARN, "circuit breaker OPEN for service: ", service_id)
end

-- 检查是否应该进入半开状态
local function check_half_open(service_id)
    local open_at      = tonumber(stats:get("open_at:" .. service_id)) or 0
    local reset_timeout = tonumber(stats:get("reset_timeout:" .. service_id)) or 30
    return (ngx.time() - open_at) >= reset_timeout
end

function Plugin:access(config)
    local route      = ngx.ctx.route
    local service_id = route and route.id or "default"

    local state = get_state(service_id)

    if state == STATE_OPEN then
        -- 检查是否可以进入半开
        if check_half_open(service_id) then
            stats:set("state:" .. service_id, STATE_HALF_OPEN)
            ngx.log(ngx.INFO, "circuit breaker HALF-OPEN for: ", service_id)
            -- 放行这一个探测请求
            ngx.ctx.cb_probe = true
            return
        end

        -- 仍然打开，直接降级
        return self:fallback(config)
    end

    -- closed 或 half_open 正常放行
    ngx.ctx.cb_service_id = service_id
end

-- 降级响应
function Plugin:fallback(config)
    local strategy = (config and config.fallback_strategy) or "reject"

    if strategy == "reject" then
        ngx.status = 503
        ngx.header["Content-Type"] = "application/json"
        ngx.header["Retry-After"]  = "30"
        ngx.say('{"code":503,"message":"service temporarily unavailable"}')
        return ngx.exit(ngx.OK)

    elseif strategy == "empty" then
        -- 返回空数据（适合读接口）
        ngx.header["Content-Type"] = "application/json"
        ngx.say(config.fallback_body or '{"code":0,"data":null}')
        return ngx.exit(ngx.OK)

    elseif strategy == "cache" then
        -- 返回缓存的旧数据（需要业务配合）
        local cache = ngx.shared.response_cache
        local cached = cache:get("fallback:" .. ngx.var.uri)
        if cached then
            ngx.header["Content-Type"] = "application/json"
            ngx.header["X-From-Cache"] = "1"
            ngx.say(cached)
            return ngx.exit(ngx.OK)
        end
        -- 缓存也没有，只能拒绝
        ngx.status = 503
        ngx.say('{"code":503,"message":"service unavailable"}')
        return ngx.exit(ngx.OK)
    end
end

-- 在 log 阶段记录请求结果，更新熔断状态
function Plugin:log(config)
    local service_id = ngx.ctx.cb_service_id
    if not service_id then return end

    local status  = ngx.status
    local success = status < 500  -- 5xx 视为失败

    record_result(service_id, success)

    local state = get_state(service_id)

    if state == STATE_HALF_OPEN then
        if success then
            -- 探测成功，关闭熔断器
            stats:set("state:" .. service_id, STATE_CLOSED)
            stats:delete("failure:" .. service_id)
            stats:delete("success:" .. service_id)
            ngx.log(ngx.INFO, "circuit breaker CLOSED for: ", service_id)
        else
            -- 探测失败，重新打开
            trip(service_id, config and config.reset_timeout)
        end
        return
    end

    -- closed 状态：检查失败率
    local threshold = (config and config.failure_threshold) or 0.5
    local min_calls = (config and config.min_calls) or 10

    local s = tonumber(stats:get("success:" .. service_id)) or 0
    local f = tonumber(stats:get("failure:" .. service_id)) or 0

    if (s + f) >= min_calls and failure_rate(service_id) >= threshold then
        trip(service_id, config and config.reset_timeout)
    end
end

return Plugin

路由配置里启用熔断：

{
  "plugins": {
    "circuit_breaker": {
      "enabled": true,
      "failure_threshold": 0.5,
      "min_calls": 10,
      "reset_timeout": 30,
      "fallback_strategy": "reject"
    }
  }
}

实战：保护后端服务

熔断器和限流搭配使用，形成双重保护：

  • 限流：控制进入网关的流量总量，防止后端被压垮
  • 熔断：检测后端健康状况，故障时主动切断，防止级联失败

当后端服务发布新版本出现 bug 导致大量 500 错误时，熔断器会在失败率达到阈值后自动触发，把流量挡在网关层，用户看到的是统一的降级响应，而不是等待超时后才得到错误。

小结

熔断器三个状态的转换逻辑：

  • 关闭 → 打开：失败率超过阈值且请求量达到最低样本数
  • 打开 → 半开：超过重置超时时间后自动探测
  • 半开 → 关闭/打开：探测请求成功则关闭，失败则重新打开

降级策略根据业务场景选择：读接口可以返回缓存或空数据，写接口通常只能拒绝并提示稍后重试。
