21 | 网关架构设计

发表于： 2026-03-30 分类于： OpenResty
字数： 1814 阅读：≈ 4分钟 浏览：

前四个阶段把 OpenResty 的语言基础、核心 API、存储集成都学完了。从这篇开始，我们用这些知识从零手写一个具备生产能力的 API 网关。

在写代码之前，先把架构想清楚。好的架构让后续扩展事半功倍，坏的架构则让每次改动都牵一发动全身。

网关的核心职责

API 网关站在客户端和后端服务之间，承担以下职责：

流量管控

  • 路由：根据请求路径、方法、Header 等分发到不同后端服务
  • 负载均衡：在多个后端实例之间分配流量
  • 限流：保护后端不被流量击垮

安全

  • 鉴权：验证请求方身份（JWT、API Key、OAuth）
  • 认证信息传递：把用户信息注入请求头传给后端
  • 防攻击：IP 黑名单、请求体大小限制

可观测性

  • 日志：记录每个请求的完整信息
  • 监控指标：请求量、延迟、错误率
  • 链路追踪：Trace ID 贯穿全链路

协议与格式转换

  • 请求/响应改写：添加/删除 Header、修改路径
  • 灰度发布：按规则把部分流量路由到新版本

插件化架构

如果把路由、鉴权、限流、日志全部写死在主逻辑里，代码很快会变成一团乱麻。插件化架构把每个功能独立成一个插件，主流程只负责按顺序调用插件。

请求到达
   │
   ▼
┌─────────────────────────────────┐
│  插件执行管道（access 阶段）      │
│  1. ip_blacklist（IP 黑名单）    │
│  2. rate_limit（限流）           │
│  3. auth（鉴权）                 │
│  4. request_rewrite（请求改写）  │
└────────────────┬────────────────┘
                 │
                 ▼
         转发到后端服务
                 │
                 ▼
┌─────────────────────────────────┐
│  插件执行管道（response 阶段）    │
│  5. response_rewrite（响应改写） │
│  6. logging（日志记录）          │
└─────────────────────────────────┘

每个插件的接口统一：

-- 插件接口约定
local Plugin = {}

-- 插件名称
Plugin.name = "my_plugin"

-- 插件优先级（数字越小越先执行）
Plugin.priority = 100

-- access 阶段（处理请求）
function Plugin:access(config)
    -- config 是该路由上此插件的配置
    -- 如需终止请求，调用 ngx.exit
end

-- response 阶段（处理响应）
function Plugin:response(config)
    -- 修改响应头、记录日志等
end

return Plugin

配置文件设计

路由配置是网关的核心数据。设计一个灵活的 JSON 格式：

{
  "routes": [
    {
      "id": "user-service",
      "path_prefix": "/api/v1/user",
      "methods": ["GET", "POST", "PUT", "DELETE"],
      "upstream": {
        "targets": [
          {"host": "user-service-1", "port": 8080, "weight": 1},
          {"host": "user-service-2", "port": 8080, "weight": 1}
        ],
        "strategy": "round_robin"
      },
      "plugins": {
        "rate_limit": {
          "enabled": true,
          "rpm": 1000,
          "burst": 200
        },
        "jwt_auth": {
          "enabled": true,
          "whitelist": ["/api/v1/user/login", "/api/v1/user/register"]
        },
        "logging": {
          "enabled": true,
          "sample_rate": 1.0
        }
      },
      "strip_prefix": true,
      "enabled": true
    },
    {
      "id": "order-service",
      "path_prefix": "/api/v1/order",
      "upstream": {
        "targets": [
          {"host": "order-service", "port": 8080, "weight": 1}
        ]
      },
      "plugins": {
        "rate_limit": {"enabled": true, "rpm": 500},
        "jwt_auth":   {"enabled": true}
      },
      "enabled": true
    }
  ],
  "global": {
    "plugins": {
      "ip_blacklist": {"enabled": true},
      "logging":      {"enabled": true, "sample_rate": 0.1}
    }
  }
}

设计原则：

  • 路由级插件：每条路由可以独立配置每个插件的参数
  • 全局插件：部分插件（如 IP 黑名单）对所有路由生效
  • 插件可独立开关：enabled 字段控制，不用删除配置
  • 向后兼容：新增字段不影响旧配置解析

nginx.conf 骨架 

worker_processes auto;
error_log logs/error.log warn;

events {
    worker_connections 10240;
    use epoll;
}

http {
    lua_package_path "$prefix/lua/?.lua;;";

    # 共享内存
    lua_shared_dict routes      5m;
    lua_shared_dict rate_limit  10m;
    lua_shared_dict blacklist   1m;
    lua_shared_dict stats       2m;

    # 启动时初始化
    init_by_lua_block {
        require "resty.core"
        -- 预加载核心模块
        require "core.router"
        require "core.plugin_runner"
    }

    init_worker_by_lua_block {
        local config_loader = require "core.config_loader"
        -- 加载路由配置
        config_loader.load()
        -- 每 30 秒刷新
        ngx.timer.every(30, function(premature)
            if premature then return end
            config_loader.reload()
        end)
    }

    server {
        listen 8080;
        server_name gateway;

        # 动态 upstream（通过 balancer_by_lua 控制）
        location @upstream {
            internal;
            proxy_pass http://dynamic_upstream;
            proxy_set_header Host              $upstream_host;
            proxy_set_header X-Real-IP         $remote_addr;
            proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
            proxy_set_header X-Request-Id      $request_id;
        }

        location / {
            set $upstream_host "";

            access_by_lua_block {
                local runner = require "core.plugin_runner"
                runner.run_access()
            }

            content_by_lua_block {
                local router = require "core.router"
                router.proxy()
            }

            log_by_lua_block {
                local runner = require "core.plugin_runner"
                runner.run_log()
            }
        }
    }
}

upstream dynamic_upstream {
    server 0.0.0.1;  -- 占位，实际 target 由 balancer_by_lua 动态设置
    balancer_by_lua_block {
        local balancer = require "core.balancer"
        balancer.balance()
    }
}

插件执行器的核心逻辑 

-- core/plugin_runner.lua
local M = {}

-- 注册的插件列表（按 priority 排序）
local plugins = {}

function M.register(plugin)
    plugins[#plugins + 1] = plugin
    -- 按优先级排序
    table.sort(plugins, function(a, b)
        return (a.priority or 100) < (b.priority or 100)
    end)
end

function M.run_access()
    local router = require "core.router"
    local route, err = router.match(ngx.var.uri, ngx.req.get_method())

    if not route then
        return require("lib.response").error(404, "no route found", 404)
    end

    ngx.ctx.route = route

    -- 执行全局插件 + 路由级插件
    for _, plugin in ipairs(plugins) do
        local plugin_config = (route.plugins or {})[plugin.name]
                           or (M.global_config.plugins or {})[plugin.name]

        if plugin_config and plugin_config.enabled then
            local ok, err = pcall(plugin.access, plugin, plugin_config)
            if not ok then
                ngx.log(ngx.ERR, "plugin [", plugin.name, "] error: ", err)
            end
        end
    end
end

function M.run_log()
    for _, plugin in ipairs(plugins) do
        if plugin.log then
            pcall(plugin.log, plugin, {})
        end
    end
end

return M

小结

网关架构的几个关键决策：

  • 插件化：每个功能独立成插件，主流程只是调度器，新增功能不影响现有代码
  • 配置驱动：路由规则和插件参数存在配置文件（或 Redis），动态加载无需重启
  • 关注点分离：core/ 是框架，plugins/ 是业务逻辑，lib/ 是工具，职责清晰
