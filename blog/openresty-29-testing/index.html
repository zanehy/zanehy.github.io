发表于： 2026-04-03 分类于： OpenResty
字数： 1655 阅读：≈ 4分钟 浏览：

写完代码，怎么验证它是对的？线上出了问题，怎么快速定位？这篇来讲 OpenResty 项目的测试和调试方法。

busted 单元测试

busted 是 Lua 生态里最成熟的测试框架，语法类似 RSpec/Jest。

安装 

luarocks install busted

基本用法 

-- spec/utils_spec.lua
local describe = describe
local it       = it
local assert   = require "luassert"

local utils = require "lib.utils"

describe("utils.slugify", function()
    it("converts spaces to hyphens", function()
        assert.equals("hello-world", utils.slugify("hello world"))
    end)

    it("lowercases the string", function()
        assert.equals("hello", utils.slugify("HELLO"))
    end)

    it("removes special characters", function()
        assert.equals("hello-world", utils.slugify("hello, world!"))
    end)
end)

describe("utils.parse_token", function()
    it("returns nil for empty string", function()
        assert.is_nil(utils.parse_token(""))
    end)

    it("extracts bearer token", function()
        local token = utils.parse_token("Bearer abc123")
        assert.equals("abc123", token)
    end)
end)

运行测试：

busted spec/          # 运行所有测试
busted spec/utils_spec.lua  # 运行单个文件
busted --verbose      # 详细输出

Mock ngx 环境

插件代码依赖 ngx 全局对象，单元测试时需要 mock：

-- spec/helpers/ngx_mock.lua
local shared_store = {}

_G.ngx = {
    -- 共享内存 mock
    shared = setmetatable({}, {
        __index = function(t, name)
            if not shared_store[name] then
                shared_store[name] = {}
                local store = shared_store[name]
                setmetatable(store, {
                    __index = {
                        get    = function(_, k) return store[k] end,
                        set    = function(_, k, v) store[k] = v; return true end,
                        incr   = function(_, k, d, init)
                            store[k] = (store[k] or init or 0) + d
                            return store[k]
                        end,
                        delete = function(_, k) store[k] = nil end,
                        add    = function(_, k, v)
                            if store[k] then return false end
                            store[k] = v; return true
                        end,
                    }
                })
            end
            return shared_store[name]
        end
    }),

    -- 请求相关
    var            = { remote_addr = "127.0.0.1", uri = "/test" },
    req            = {
        get_headers  = function() return {} end,
        get_uri_args = function() return {} end,
        set_header   = function() end,
        clear_header = function() end,
    },
    header         = {},
    status         = 200,
    ctx            = {},

    -- 常量
    OK     = 200,
    ERR    = 500,
    WARN   = 301,
    INFO   = 302,
    DEBUG  = 303,

    -- 函数
    log    = function() end,
    time   = function() return os.time() end,
    now    = function() return os.time() end,
    sleep  = function() end,
    exit   = function(code) error("ngx.exit:" .. (code or 0)) end,
    say    = function(s) io.write(tostring(s) .. "\n") end,
}

测试里引入这个 mock：

-- spec/plugins/jwt_auth_spec.lua
require "spec.helpers.ngx_mock"

local jwt_auth = require "plugins.jwt_auth"
local jwt      = require "resty.jwt"

describe("jwt_auth plugin", function()
    local secret = "test_secret"

    local function make_token(payload)
        return jwt:sign(secret, {
            header  = { typ = "JWT", alg = "HS256" },
            payload = payload,
        })
    end

    it("rejects request without token", function()
        ngx.req.get_headers = function() return {} end
        assert.error_matches(function()
            jwt_auth:access({ secret = secret })
        end, "ngx.exit:401")
    end)

    it("accepts valid token", function()
        local token = make_token({ sub = "123", exp = os.time() + 3600 })
        ngx.req.get_headers = function()
            return { authorization = "Bearer " .. token }
        end
        assert.has_no_error(function()
            jwt_auth:access({ secret = secret })
        end)
        assert.equals("123", ngx.ctx.user.id)
    end)

    it("rejects expired token", function()
        local token = make_token({ sub = "123", exp = os.time() - 1 })
        ngx.req.get_headers = function()
            return { authorization = "Bearer " .. token }
        end
        assert.error_matches(function()
            jwt_auth:access({ secret = secret })
        end, "ngx.exit:401")
    end)
end)

Test::Nginx 集成测试

单元测试验证单个模块，集成测试验证整个请求链路。Test::Nginx 是 OpenResty 官方的集成测试框架，用 Perl 编写，通过启动真实的 Nginx 进程来测试。

安装 

cpan Test::Nginx
# 或
cpanm Test::Nginx

测试文件格式 

# t/jwt_auth.t
use Test::Nginx::Socket 'no_plan';

# 运行测试前执行的 Perl 代码
our $secret = "test_secret";

run_tests();

__DATA__

=== TEST 1: missing token returns 401
--- config
    location /api/test {
        access_by_lua_block {
            -- 配置测试用的 secret
            package.loaded["resty.jwt_secret"] = "test_secret"
            require("plugins.jwt_auth"):access({
                secret = "test_secret",
                enabled = true,
            })
        }
        content_by_lua_block {
            ngx.say("ok")
        }
    }
--- request
GET /api/test
--- response_body_like: missing token
--- error_code: 401

=== TEST 2: valid token passes
--- config
    location /api/test {
        access_by_lua_block {
            require("plugins.jwt_auth"):access({
                secret  = "test_secret",
                enabled = true,
            })
        }
        content_by_lua_block {
            ngx.say("user:", ngx.ctx.user.id)
        }
    }
--- request eval
use POSIX qw(time);
my $exp = time() + 3600;
# 生成 JWT（简化，实际用 Perl 的 JWT 库）
"GET /api/test\nAuthorization: Bearer eyJ..."
--- response_body_like: user:123
--- error_code: 200

运行测试 

prove -r t/           # 运行所有测试
prove t/jwt_auth.t    # 运行单个文件
prove -v t/           # 详细输出

本地调试技巧

1. 合理使用 ngx.log 

-- 开发时用 DEBUG 级别记录详细信息
ngx.log(ngx.DEBUG, "request args: ", require("cjson").encode(ngx.req.get_uri_args()))
ngx.log(ngx.DEBUG, "route matched: ", ngx.ctx.route and ngx.ctx.route.id)

-- 注意：字符串拼接在 DEBUG 日志里也有开销
-- 可以先判断日志级别（OpenResty 没有直接的 is_debug_enabled，但可以用 pcall 技巧）

在 nginx.conf 里把日志级别调到 debug：

error_log logs/error.log debug;

生产环境记得改回 warn 或 error，debug 级别日志量极大。

2. 实时查看日志 

# 实时追踪错误日志
tail -f logs/error.log

# 过滤特定内容
tail -f logs/error.log | grep "jwt_auth"

# 查看最近的错误
tail -100 logs/error.log | grep "\[error\]"

3. 使用 ngx.say 临时 dump

调试复杂数据结构时，临时用 ngx.say 输出：

-- 临时调试，上线前记得删除！
local cjson = require "cjson"
ngx.say("DEBUG: ", cjson.encode({
    route   = ngx.ctx.route,
    user    = ngx.ctx.user,
    headers = ngx.req.get_headers(),
}))
return ngx.exit(200)

常见报错排查手册

attempt to index a nil value

最常见的报错，通常是某个变量没有判断 nil：

-- 错误示例
local user = ngx.ctx.user
ngx.say(user.name)  -- 如果 user 是 nil 就崩

-- 正确
local user = ngx.ctx.user
if not user then
    return ngx.exit(401)
end
ngx.say(user.name)

cosocket API disabled in the current context

在不允许使用 cosocket 的阶段（如 init_by_lua、log_by_lua）尝试做网络操作：

-- 错误：log_by_lua 里不能用 cosocket
log_by_lua_block {
    local red = redis:new()
    red:connect(...)  -- 报错！
}

-- 正确：用 timer 异步处理
log_by_lua_block {
    ngx.timer.at(0, function(premature)
        local red = redis:new()
        red:connect(...)  -- OK
    end)
}

no resolver defined to resolve

在 Lua 里用域名连接外部服务，但没有配置 DNS resolver：

# nginx.conf 里加上
resolver 8.8.8.8 valid=30s;
resolver_timeout 5s;

lua_max_running_timers are not enough

定时器数量超出限制，调大配置：

lua_max_pending_timers 2048;
lua_max_running_timers 256;

bad argument #1 to 'encode' (table expected, got nil)

cjson.encode 传了 nil，在 encode 之前检查：

local data = get_data()
if not data then
    ngx.say("null")
    return
end
ngx.say(cjson.encode(data))

connection refused 连接后端失败

排查步骤：

  1. 确认后端服务是否在运行：curl http://backend:8080/health
  2. 确认网络连通性：telnet backend 8080
  3. 查看 nginx 错误日志里的详细错误信息
  4. 检查连接池是否耗尽（日志里有 no keepalive peer available）

attempt to perform arithmetic on a string value

从 ngx.req.get_uri_args() 或 Redis 拿到的值是字符串，要做数学运算必须先转换：

local page = tonumber(ngx.req.get_uri_args().page) or 1
local offset = (page - 1) * 10

小结

测试和调试的工具链：

  • busted：单元测试，测单个模块，需要 mock ngx 环境
  • Test::Nginx：集成测试，启动真实 Nginx，测完整请求链路
  • ngx.log + tail -f：线上问题排查的基本手段
  • 常见报错基本都有规律，遇到一次记住一次
