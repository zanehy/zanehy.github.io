20 | 日志系统设计

发表于： 2026-03-29 分类于： OpenResty
字数： 1515 阅读：≈ 4分钟 浏览：

日志是系统的眼睛。一个好的日志系统，不只是把信息写到文件里，还要考虑：格式是否方便机器解析、是否影响请求性能、如何与下游日志平台对接。这篇来系统地聊 OpenResty 的日志设计。

结构化日志输出

传统的日志是人可读的文本，比如 Nginx 默认的 combined 格式：

192.168.1.1 - - [10/Mar/2026:10:00:00 +0800] "GET /api/users HTTP/1.1" 200 1234

这种格式对人友好，但对机器不友好——解析时需要正则匹配，字段顺序固定，扩展字段麻烦。

结构化日志用 JSON 格式，每行一个完整的日志对象：

{"time":"2026-03-10T10:00:00+08:00","method":"GET","uri":"/api/users","status":200,"body_bytes":1234,"request_time":0.023,"remote_addr":"192.168.1.1","user_id":"123","trace_id":"abc-xyz"}

字段清晰，扩展方便，直接被 Elasticsearch、Loki 等平台摄取。

自定义 access_log 格式

Nginx 的 log_format 指令支持自定义日志格式。结合 OpenResty 的变量，可以输出结构化的 JSON 日志：

http {
    log_format json_log escape=json
        '{'
            '"time":"$time_iso8601",'
            '"remote_addr":"$remote_addr",'
            '"method":"$request_method",'
            '"uri":"$uri",'
            '"query":"$query_string",'
            '"status":$status,'
            '"body_bytes":$body_bytes_sent,'
            '"request_time":$request_time,'
            '"upstream_time":"$upstream_response_time",'
            '"http_referer":"$http_referer",'
            '"user_agent":"$http_user_agent",'
            '"x_forwarded_for":"$http_x_forwarded_for"'
        '}';

    access_log /var/log/nginx/access.log json_log;
}

escape=json 是 Nginx 1.11.8+ 引入的选项，会自动对字段值里的特殊字符（引号、反斜杠等）做 JSON 转义，避免日志格式损坏。

添加业务字段

Nginx 变量只能覆盖基础信息，业务字段（用户 ID、Trace ID 等）需要在 Lua 里写入变量：

http {
    log_format json_log escape=json
        '{'
            '"time":"$time_iso8601",'
            '"uri":"$uri",'
            '"status":$status,'
            '"user_id":"$user_id",'
            '"trace_id":"$trace_id",'
            '"request_time":$request_time'
        '}';

    server {
        set $user_id  "";
        set $trace_id "";

        location /api/ {
            access_by_lua_block {
                -- 解析 Token，写入变量
                local user = verify_and_get_user()
                if user then
                    ngx.var.user_id = tostring(user.id)
                end

                -- 生成或透传 Trace ID
                local trace_id = ngx.req.get_headers()["x-trace-id"]
                if not trace_id then
                    trace_id = ngx.md5(tostring(ngx.now()) .. ngx.var.remote_addr)
                end
                ngx.var.trace_id = trace_id
                ngx.req.set_header("X-Trace-Id", trace_id)
            }

            content_by_lua_file lua/handlers/api.lua;
        }
    }
}

用 log_by_lua 记录更丰富的日志

access_log 只能记录 Nginx 变量，如果想记录请求 body、响应体片段、自定义指标等，需要在 log_by_lua 阶段自己写：

log_by_lua_block {
    local cjson = require "cjson"

    local log_entry = {
        time         = ngx.now(),
        method       = ngx.var.request_method,
        uri          = ngx.var.uri,
        status       = ngx.status,
        request_time = ngx.now() - (ngx.ctx.start_time or ngx.now()),
        remote_addr  = ngx.var.remote_addr,
        user_id      = ngx.ctx.user and ngx.ctx.user.id or nil,
        trace_id     = ngx.var.trace_id,
    }

    -- 写入共享内存队列（不阻塞请求）
    local queue = ngx.shared.log_queue
    local seq   = queue:incr("__seq", 1, 0)
    queue:set("log:" .. seq, cjson.encode(log_entry), 30)
}

异步日志上报

直接在请求处理里写日志到远端（Kafka、HTTP 接口）会增加请求延迟。正确的方式是写入内存队列，由后台定时器批量上报。

完整的异步日志系统 

http {
    lua_shared_dict log_queue 20m;

    init_worker_by_lua_block {
        -- 每 3 秒批量上报一次
        ngx.timer.every(3, function(premature)
            if premature then return end

            local queue   = ngx.shared.log_queue
            local cjson   = require "cjson"
            local httpc   = require("resty.http").new()

            -- 读取积压的日志
            local seq_max = tonumber(queue:get("__seq")) or 0
            local seq_last = tonumber(queue:get("__seq_sent")) or 0

            if seq_max <= seq_last then return end

            local batch = {}
            local new_last = seq_last
            for i = seq_last + 1, math.min(seq_last + 200, seq_max) do
                local entry = queue:get("log:" .. i)
                if entry then
                    batch[#batch + 1] = entry
                    queue:delete("log:" .. i)
                    new_last = i
                end
            end

            if #batch == 0 then return end

            -- 上报到日志服务
            httpc:set_timeouts(2000, 5000, 10000)
            local ok, err = httpc:request_uri("http://log-collector.internal/batch", {
                method  = "POST",
                body    = "[" .. table.concat(batch, ",") .. "]",
                headers = { ["Content-Type"] = "application/json" },
                keepalive_timeout = 30000,
                keepalive_pool    = 5,
            })

            if ok and ok.status < 300 then
                queue:set("__seq_sent", new_last)
            else
                ngx.log(ngx.ERR, "log upload failed: ", err or ok.status)
            end
        end)
    }
}

对接 Kafka

Kafka 是大规模日志收集的常见选择。用 lua-resty-kafka 库直接从 OpenResty 写入 Kafka：

opm get doujiang24/lua-resty-kafka

-- lib/kafka_logger.lua
local producer = require "resty.kafka.producer"
local cjson    = require "cjson"

local M = {}

-- Kafka broker 地址
local broker_list = {
    { host = "kafka1.internal", port = 9092 },
    { host = "kafka2.internal", port = 9092 },
}

-- 异步 producer（不等待确认，最高吞吐）
local kafka_producer

local function get_producer()
    if kafka_producer then return kafka_producer end

    kafka_producer = producer:new(broker_list, {
        producer_type = "async",
        flush_time    = 1000,     -- 每秒 flush 一次
        batch_num     = 200,      -- 每批最多 200 条
    })
    return kafka_producer
end

function M.send(log_entry)
    local p = get_producer()
    local ok, err = p:send("nginx-access-logs", nil, cjson.encode(log_entry))
    if not ok then
        ngx.log(ngx.ERR, "kafka send failed: ", err)
    end
end

return M

在 log_by_lua 里使用：

log_by_lua_block {
    local kafka = require "lib.kafka_logger"
    kafka.send({
        time        = ngx.now(),
        uri         = ngx.var.uri,
        status      = ngx.status,
        user_id     = ngx.ctx.user and ngx.ctx.user.id,
        trace_id    = ngx.var.trace_id,
        server_name = ngx.var.server_name,
    })
}

对接 Elasticsearch

如果日志量不大，可以跳过 Kafka，直接写入 ES 的 Bulk API：

-- 定时器里批量写入 ES
local function flush_to_es(logs)
    local httpc = require("resty.http").new()
    httpc:set_timeouts(3000, 5000, 15000)

    -- ES Bulk API 格式：每条记录两行（action + data）
    local lines = {}
    local index = "nginx-logs-" .. os.date("%Y.%m.%d")
    for _, log_str in ipairs(logs) do
        lines[#lines + 1] = '{"index":{"_index":"' .. index .. '"}}'
        lines[#lines + 1] = log_str
    end
    lines[#lines + 1] = ""  -- 末尾换行

    local res, err = httpc:request_uri("http://elasticsearch.internal:9200/_bulk", {
        method  = "POST",
        body    = table.concat(lines, "\n"),
        headers = { ["Content-Type"] = "application/x-ndjson" },
        keepalive_timeout = 30000,
        keepalive_pool    = 3,
    })

    if not res or res.status >= 300 then
        ngx.log(ngx.ERR, "es bulk failed: ", err or res.status)
    end
end

日志采样

高并发场景下，每条请求都记录详细日志会产生海量数据。可以按比例采样：

log_by_lua_block {
    -- 只记录 10% 的请求（随机采样）
    if math.random() > 0.1 then return end

    -- 或者：错误请求 100% 记录，正常请求 5% 采样
    if ngx.status < 400 and math.random() > 0.05 then return end

    -- 记录日志...
}

小结

一个完整的 OpenResty 日志系统包含这几个层次：

  • 格式：用 JSON 结构化日志，方便机器解析和平台摄取
  • 字段：通过 log_format 和 Lua 变量注入业务字段（用户 ID、Trace ID）
  • 性能：用 log_by_lua 写内存队列，后台定时器批量上报，不阻塞请求
  • 下游：Kafka 适合大规模，ES 适合中等规模；两者都通过批量写入降低压力
  • 成本：采样降低日志量，按错误率动态调整采样率

到这里，第四阶段（中间件与存储）全部讲完。下一篇进入第五阶段，开始手写 API 网关——从架构设计开始。
