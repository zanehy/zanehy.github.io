25 | 限流模块

发表于： 2026-04-01 分类于： OpenResty
字数： 2008 阅读：≈ 5分钟 浏览：

鉴权解决了"谁能访问"的问题，限流解决"能访问多少次"的问题。没有限流的 API 网关是不完整的——不管是防止恶意刷接口，还是保护后端服务不被突发流量击垮，限流都是不可或缺的。

三种限流算法

固定窗口

把时间切成固定大小的窗口（如每分钟），统计窗口内的请求数，超过阈值则拒绝。

时间轴: |--- 0:00 ~ 0:01 ---|--- 0:01 ~ 0:02 ---|
         [请求数: 95]          [请求数: 3]
                    ↑ 在 0:00:59 发 5 个请求不会超限（95+5=100）
                      但 0:01:00 再发 5 个也不超限（3+5=8）
                      窗口切换后计数清零，临界点可能出现 2 倍突刺

优点：实现简单，Redis 里一个 INCR + EXPIRE 就够。 缺点：窗口切换时的临界突刺——在 0:00:59 发 100 个请求，0:01:00 再发 100 个，1 秒内实际处理了 200 个请求。

滑动窗口

用精确的时间范围（当前时间往前 N 秒）统计请求数，没有固定边界，消除了临界突刺。

当前时间: 0:01:30，窗口大小: 60s
统计范围: 0:00:30 ~ 0:01:30 内的所有请求

Redis 实现：用 Sorted Set，以时间戳为 score，每次请求 ZADD，统计时 ZCOUNT 指定时间范围。

优点：没有窗口边界问题，限流更平滑。 缺点：每个请求写一条记录，高并发时内存占用高。

令牌桶

有一个固定容量的桶，系统以固定速率往桶里放令牌。每个请求消耗一个令牌，桶空了就拒绝。令牌可以积累，所以允许一定程度的突发流量。

桶容量: 100，填充速率: 10 个/秒
桶里初始 100 个令牌
→ 可以瞬间处理 100 个请求（消耗所有令牌）
→ 之后每秒最多处理 10 个（稳定速率）
→ 如果某秒只来了 5 个请求，桶里积累 5 个，供后续突发使用

优点：允许突发，限制长期平均速率，对用户体验友好。 缺点：实现稍复杂，需要计算令牌数量。

固定窗口限流实现

最简单的实现，适合大多数场景：

-- Redis Lua 脚本（原子操作）
local FIXED_WINDOW_SCRIPT = [[
local key     = KEYS[1]
local limit   = tonumber(ARGV[1])
local window  = tonumber(ARGV[2])  -- 窗口大小（秒）

local current = tonumber(redis.call("GET", key) or "0")
if current >= limit then
    return 0  -- 超限
end

local new_val = redis.call("INCR", key)
if new_val == 1 then
    -- 第一次设置过期时间
    redis.call("EXPIRE", key, window)
end
return 1  -- 允许
]]

local function fixed_window_check(red, key, limit, window)
    local res, err = red:eval(FIXED_WINDOW_SCRIPT, 1, key, limit, window)
    if not res then
        ngx.log(ngx.ERR, "rate limit script failed: ", err)
        return true  -- Redis 出错时放行（可用性优先）
    end
    return res == 1
end

滑动窗口限流实现 

local SLIDING_WINDOW_SCRIPT = [[
local key        = KEYS[1]
local now        = tonumber(ARGV[1])  -- 当前时间（毫秒时间戳）
local window_ms  = tonumber(ARGV[2])  -- 窗口大小（毫秒）
local limit      = tonumber(ARGV[3])

local window_start = now - window_ms

-- 删除窗口外的旧记录
redis.call("ZREMRANGEBYSCORE", key, "-inf", window_start)

-- 统计窗口内的请求数
local count = tonumber(redis.call("ZCARD", key))

if count >= limit then
    return 0  -- 超限
end

-- 记录本次请求（score 和 member 都用时间戳，加随机后缀避免重复）
redis.call("ZADD", key, now, now .. math.random(1, 1000000))
redis.call("PEXPIRE", key, window_ms)
return 1
]]

local function sliding_window_check(red, key, limit, window_seconds)
    local now = math.floor(ngx.now() * 1000)  -- 毫秒
    local res, err = red:eval(
        SLIDING_WINDOW_SCRIPT, 1,
        key, now, window_seconds * 1000, limit
    )
    if not res then return true end
    return res == 1
end

令牌桶限流实现 

local TOKEN_BUCKET_SCRIPT = [[
local key        = KEYS[1]
local capacity   = tonumber(ARGV[1])  -- 桶容量
local rate       = tonumber(ARGV[2])  -- 填充速率（个/秒）
local now        = tonumber(ARGV[3])  -- 当前时间（秒，支持小数）
local requested  = tonumber(ARGV[4])  -- 本次消耗令牌数（通常为 1）

local last_time  = tonumber(redis.call("HGET", key, "last_time") or now)
local tokens     = tonumber(redis.call("HGET", key, "tokens") or capacity)

-- 计算新增令牌数
local delta  = math.max(0, now - last_time)
local filled = math.min(capacity, tokens + delta * rate)

if filled < requested then
    -- 令牌不足
    redis.call("HMSET", key, "last_time", now, "tokens", filled)
    redis.call("EXPIRE", key, math.ceil(capacity / rate) + 1)
    return 0
end

-- 消耗令牌
redis.call("HMSET", key, "last_time", now, "tokens", filled - requested)
redis.call("EXPIRE", key, math.ceil(capacity / rate) + 1)
return 1
]]

local function token_bucket_check(red, key, capacity, rate)
    local now = ngx.now()
    local res, err = red:eval(
        TOKEN_BUCKET_SCRIPT, 1,
        key, capacity, rate, now, 1
    )
    if not res then return true end
    return res == 1
end

完整的限流插件 

-- plugins/rate_limit.lua
local redis    = require "resty.redis"
local response = require "lib.response"

local Plugin = {}
Plugin.name     = "rate_limit"
Plugin.priority = 10  -- 限流在鉴权之前执行

local function get_redis()
    local red = redis:new()
    red:set_timeouts(200, 200, 200)  -- 限流要快，超时设短
    local ok, err = red:connect("127.0.0.1", 6379)
    if not ok then return nil, err end
    return red
end

-- 构造限流 key
local function build_key(dimension, config)
    local prefix = "rl:" .. (config.route_id or "global") .. ":"

    if dimension == "ip" then
        return prefix .. "ip:" .. ngx.var.remote_addr
    elseif dimension == "user" then
        local user = ngx.ctx.user
        if not user then return nil end
        return prefix .. "user:" .. tostring(user.id)
    elseif dimension == "global" then
        return prefix .. "global"
    end
    return nil
end

function Plugin:access(config)
    local red, err = get_redis()
    if not red then
        ngx.log(ngx.WARN, "rate limit redis unavailable: ", err)
        return  -- Redis 不可用时放行
    end

    local window  = config.window or 60    -- 默认 60 秒窗口
    local allowed = true

    -- 多维度限流：IP > 用户 > 全局
    local checks = {
        { dimension = "ip",     limit = config.ip_limit  },
        { dimension = "user",   limit = config.user_limit },
        { dimension = "global", limit = config.rpm        },
    }

    for _, check in ipairs(checks) do
        if check.limit then
            local key = build_key(check.dimension, config)
            if key then
                local ok = fixed_window_check(red, key, check.limit, window)
                if not ok then
                    allowed = false
                    ngx.ctx.rate_limit_dimension = check.dimension
                    break
                end
            end
        end
    end

    red:set_keepalive(10000, 50)

    if not allowed then
        -- 设置标准限流响应头
        ngx.header["Retry-After"]       = tostring(window)
        ngx.header["X-RateLimit-Limit"] = tostring(config.rpm or 0)
        ngx.header["X-RateLimit-Reset"] = tostring(
            math.ceil(ngx.time() / window) * window
        )

        local msg = config.deny_message or "rate limit exceeded"
        return response.error(429, msg, 429)
    end
end

return Plugin

优雅降级响应

被限流后，不应该直接返回生硬的错误，而是给客户端足够的信息：

-- 标准的限流响应
ngx.status = 429
ngx.header["Content-Type"]       = "application/json"
ngx.header["Retry-After"]        = "60"          -- 60 秒后重试
ngx.header["X-RateLimit-Limit"]  = "1000"        -- 限额
ngx.header["X-RateLimit-Reset"]  = tostring(next_window_start)  -- 下一个窗口开始时间
ngx.say(cjson.encode({
    code    = 429,
    message = "Too Many Requests",
    retry_after = 60,
}))

部分场景下，限流不是拒绝而是降级——比如返回缓存数据、返回简化版响应：

function Plugin:access(config)
    -- ... 检查限流 ...
    if not allowed then
        if config.fallback == "cache" then
            -- 返回缓存数据
            local cached = ngx.shared.response_cache:get(ngx.var.uri)
            if cached then
                ngx.header["X-Cache"]  = "HIT"
                ngx.header["X-Throttled"] = "true"
                ngx.say(cached)
                return ngx.exit(ngx.OK)
            end
        end
        -- 无缓存则返回 429
        return response.error(429, "rate limit exceeded", 429)
    end
end

  "id": "order-service",

      "rpm": 1000,
      "ip_limit": 100,

      "window": 60,
      "deny_message": "request too frequent, please try again later"
    }
  }
}

三个维度同时生效：

  • 全局每分钟不超过 1000 次
  • 单 IP 每分钟不超过 100 次
  • 单用户每分钟不超过 200 次

任意一个维度触发限流，请求即被拦截。

小结

三种算法各有适用场景：

  • 固定窗口：实现简单，有临界突刺问题，适合对精确性要求不高的场景
  • 滑动窗口：更精确，内存占用高，适合需要严格限流的场景
  • 令牌桶：允许突发，对用户友好，适合 API 对外开放场景

限流插件的设计要点：

  • Redis 不可用时放行（可用性 > 限流精确性，避免 Redis 故障导致业务中断）
  • 支持多维度（IP、用户、全局），维度越细控制越精准
  • 响应头里提供 Retry-After 和 X-RateLimit-*，让客户端能合理重试
