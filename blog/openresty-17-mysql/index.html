17 | 连接 MySQL

发表于： 2026-03-25 分类于： OpenResty
字数： 1639 阅读：≈ 4分钟 浏览：

Redis 适合存临时数据和缓存，但涉及结构化数据、需要事务保证、需要复杂查询的场景，还是得上 MySQL。这篇来讲 OpenResty 怎么连接 MySQL。

lua-resty-mysql 基本使用

lua-resty-mysql 同样是 OpenResty 官方库，基于 cosocket，完全非阻塞。OpenResty 发行版内置，直接 require 即可。

连接 

local mysql = require "resty.mysql"

local db, err = mysql:new()
if not db then
    ngx.log(ngx.ERR, "failed to create mysql: ", err)
    return ngx.exit(500)
end

-- 设置超时
db:set_timeout(1000)  -- 单个超时值，单位毫秒

-- 连接数据库
local ok, err, errcode, sqlstate = db:connect({
    host     = "127.0.0.1",
    port     = 3306,
    database = "myapp",
    user     = "appuser",
    password = "secret",
    charset  = "utf8mb4",
    max_packet_size = 1024 * 1024,  -- 1MB
})

if not ok then
    ngx.log(ngx.ERR, "mysql connect failed: ", err, ", code: ", errcode)
    return ngx.exit(500)
end

执行查询 

-- 查询返回结果集
local res, err, errcode, sqlstate = db:query("SELECT id, name, email FROM users LIMIT 10")

if not res then
    ngx.log(ngx.ERR, "query failed: ", err)
    return
end

-- res 是一个数组，每个元素是一行数据（table，键为列名）
for _, row in ipairs(res) do
    ngx.say(row.id, " | ", row.name, " | ", row.email)
end

-- 执行 INSERT / UPDATE / DELETE
local res, err = db:query("INSERT INTO logs (uri, status) VALUES ('/api', 200)")
if res then
    ngx.say("inserted id: ", res.insert_id)
    ngx.say("affected rows: ", res.affected_rows)
end

连接池

和 Redis 一样，用完连接必须归还连接池：

-- 放回连接池：空闲超时 10 秒，池大小 50
local ok, err = db:set_keepalive(10000, 50)
if not ok then
    ngx.log(ngx.WARN, "set keepalive failed: ", err)
    db:close()
end

预编译语句防 SQL 注入

直接拼接 SQL 字符串是危险的：

-- 危险！用户可以注入 SQL
local name = ngx.var.arg_name   -- 假设输入 "' OR '1'='1"
local sql = "SELECT * FROM users WHERE name = '" .. name .. "'"
db:query(sql)  -- 逻辑被篡改！

正确做法是转义或使用参数化查询。lua-resty-mysql 提供了转义函数：

-- 方法一：手动转义
local name = ngx.var.arg_name
local escaped_name = db:escape_literal(name)
-- escape_literal 加引号，例如 'alice'
-- escape_identifier 用于表名/列名，加反引号，例如 `username`

local sql = "SELECT * FROM users WHERE name = " .. escaped_name
local res = db:query(sql)

不过更推荐的方式是封装一个占位符替换函数，写法更清晰：

-- 简单的参数化查询封装
local function query(db, sql, ...)
    local args = {...}
    local i = 0
    local safe_sql = sql:gsub("%?", function()
        i = i + 1
        local val = args[i]
        if val == nil then
            return "NULL"
        elseif type(val) == "number" then
            return tostring(val)
        else
            return db:escape_literal(tostring(val))
        end
    end)
    return db:query(safe_sql)
end

-- 使用
local res, err = query(db,
    "SELECT * FROM users WHERE name = ? AND age > ?",
    ngx.var.arg_name,
    18
)

使用 Prepared Statement

lua-resty-mysql 也支持 MySQL 协议层面的 Prepared Statement：

-- 准备语句
local stmt, err = db:prepare("SELECT id, name FROM users WHERE email = ? AND status = ?")
if not stmt then
    ngx.log(ngx.ERR, "prepare failed: ", err)
    return
end

-- 执行（参数在协议层绑定，完全规避注入）
local res, err = stmt:execute("alice@example.com", 1)
if not res then
    ngx.log(ngx.ERR, "execute failed: ", err)
    return
end

for _, row in ipairs(res) do
    ngx.say(row.id, " ", row.name)
end

stmt:close()

事务处理 

-- 开启事务
local ok, err = db:query("START TRANSACTION")
if not ok then
    ngx.log(ngx.ERR, "start transaction failed: ", err)
    return ngx.exit(500)
end

-- 执行多条语句
local res1, err1 = db:query(
    "UPDATE accounts SET balance = balance - 100 WHERE id = 1"
)
local res2, err2 = db:query(
    "UPDATE accounts SET balance = balance + 100 WHERE id = 2"
)

-- 检查是否全部成功
if not res1 or not res2 then
    -- 回滚
    db:query("ROLLBACK")
    ngx.log(ngx.ERR, "transaction failed, rolled back")
    db:set_keepalive(10000, 50)
    return ngx.exit(500)
end

-- 提交
local ok, err = db:query("COMMIT")
if not ok then
    ngx.log(ngx.ERR, "commit failed: ", err)
    db:set_keepalive(10000, 50)
    return ngx.exit(500)
end

db:set_keepalive(10000, 50)
ngx.say("transfer done")

注意：事务期间的连接不能放回连接池（因为事务状态绑定在连接上），必须等事务结束（COMMIT 或 ROLLBACK）后再调用 set_keepalive。

实战：从数据库动态读取路由配置

这是 API 网关里的常见需求：路由规则存在数据库里，不重启 nginx 就能更新。

数据库表设计 

CREATE TABLE routes (
    id          INT PRIMARY KEY AUTO_INCREMENT,
    path_prefix VARCHAR(200) NOT NULL,
    upstream    VARCHAR(200) NOT NULL,
    strip_prefix TINYINT DEFAULT 0,
    enabled     TINYINT DEFAULT 1,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP
);

INSERT INTO routes VALUES
(1, '/api/user',    'http://user-service:8080',    1, 1, NOW()),
(2, '/api/order',   'http://order-service:8080',   1, 1, NOW()),
(3, '/api/product', 'http://product-service:8080', 1, 1, NOW());

路由加载模块 

-- lib/route_loader.lua
local mysql = require "resty.mysql"
local cjson = require "cjson"
local M = {}

local function get_db()
    local db = mysql:new()
    db:set_timeout(2000)
    local ok, err = db:connect({
        host     = "127.0.0.1",
        port     = 3306,
        database = "gateway",
        user     = "gateway",
        password = "secret",
        charset  = "utf8mb4",
    })
    if not ok then return nil, err end
    return db
end

-- 从数据库加载路由，写入共享内存
function M.reload()
    local db, err = get_db()
    if not db then
        return false, "db connect failed: " .. (err or "")
    end

    local res, err = db:query(
        "SELECT path_prefix, upstream, strip_prefix FROM routes WHERE enabled = 1 ORDER BY LENGTH(path_prefix) DESC"
    )
    db:set_keepalive(10000, 10)

    if not res then
        return false, "query failed: " .. (err or "")
    end

    local cache = ngx.shared.routes
    cache:set("routes", cjson.encode(res))
    cache:set("routes_loaded_at", ngx.time())

    ngx.log(ngx.INFO, "routes reloaded: ", #res, " rules")
    return true
end

-- 从共享内存读取路由，匹配当前 URI
function M.match(uri)
    local cache = ngx.shared.routes
    local routes_json = cache:get("routes")
    if not routes_json then
        return nil, "routes not loaded"
    end

    local routes = cjson.decode(routes_json)
    for _, route in ipairs(routes) do
        if uri:sub(1, #route.path_prefix) == route.path_prefix then
            return route
        end
    end
    return nil
end

return M

nginx.conf 集成 

http {
    lua_shared_dict routes 1m;

    init_worker_by_lua_block {
        local loader = require "lib.route_loader"

        -- 启动时加载路由
        loader.reload()

        -- 每 30 秒刷新一次
        ngx.timer.every(30, function(premature)
            if premature then return end
            loader.reload()
        end)
    }

    server {
        location / {
            content_by_lua_block {
                local loader = require "lib.route_loader"
                local uri = ngx.var.uri

                local route, err = loader.match(uri)
                if not route then
                    ngx.status = 404
                    ngx.say('{"code":404,"message":"no route found"}')
                    return
                end

                -- 动态设置 upstream（实际项目用 ngx.balancer 更完整）
                local target_uri = uri
                if route.strip_prefix == 1 then
                    target_uri = uri:sub(#route.path_prefix + 1) or "/"
                end

                ngx.var.proxy_target = route.upstream .. target_uri
                ngx.exec("@proxy")
            }
        }

        location @proxy {
            internal;
            proxy_pass $proxy_target;
        }
    }
}

路由数据每 30 秒从 MySQL 刷新一次，存在共享内存里供所有请求使用。修改路由规则后，最多 30 秒生效，无需重启 nginx。

小结

lua-resty-mysql 的使用要点：

  • 连接参数包括 charset，推荐 utf8mb4
  • 永远不要拼接用户输入进 SQL，用 escape_literal 或封装占位符
  • 事务期间的连接不能放回连接池，必须等事务结束
  • 动态路由等配置类数据，用"定时器刷新 + 共享内存缓存"的模式，兼顾实时性和性能
