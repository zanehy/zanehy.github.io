23 | 鉴权模块：JWT 认证

发表于： 2026-03-31 分类于： OpenResty
字数： 1706 阅读：≈ 4分钟 浏览：

路由模块搭好了，下一步是鉴权。没有通过鉴权的请求，不应该打到后端服务。这篇实现 JWT 鉴权插件。

JWT 原理与结构

JWT（JSON Web Token）是一种自包含的令牌格式。一个 JWT 由三部分组成，用 . 分隔：

  • Header（第一段）：Base64 编码的 JSON，包含算法类型（如 HS256）
  • Payload（第二段）：Base64 编码的 JSON，包含声明（Claims），如用户 ID、过期时间
  • Signature（第三段）：用密钥对 Header + Payload 做 HMAC 签名，防篡改

标准 Payload 字段：

字段含义
subSubject，通常是用户 ID
expExpiration，过期时间（Unix 时间戳）
iatIssued At，签发时间
jtiJWT ID，唯一标识，用于撤销
issIssuer，签发者

JWT 的验证逻辑：用同样的密钥重新计算签名，与 Token 中的签名对比——一致则说明 Token 未被篡改，再检查 exp 是否过期。

实现 JWT 鉴权插件 

-- plugins/jwt_auth.lua
local jwt      = require "resty.jwt"
local cjson    = require "cjson"
local response = require "lib.response"

local Plugin = {}
Plugin.name     = "jwt_auth"
Plugin.priority = 20   -- 鉴权在限流之后执行

-- 从请求头提取 Token
local function extract_token()
    local auth_header = ngx.req.get_headers()["authorization"] or ""

    -- Bearer Token
    local token = auth_header:match("^[Bb]earer%s+(.+)$")
    if token then return token end

    -- 也支持 query string 传 token（某些场景需要）
    local args = ngx.req.get_uri_args()
    return args.access_token
end

-- 检查路径是否在白名单
local function is_whitelisted(uri, whitelist)
    if not whitelist then return false end
    for _, path in ipairs(whitelist) do
        if uri == path or uri:sub(1, #path) == path then
            return true
        end
    end
    return false
end

function Plugin:access(config)
    local uri = ngx.var.uri

    -- 白名单路由直接放行
    if is_whitelisted(uri, config.whitelist) then
        return
    end

    -- 提取 Token
    local token = extract_token()
    if not token then
        return response.error(401, "missing token", 401)
    end

    -- 验证 Token
    local secret = config.secret or os.getenv("JWT_SECRET") or "default_secret"
    local jwt_obj = jwt:verify(secret, token)

    if not jwt_obj.verified then
        local reason = jwt_obj.reason or "invalid token"
        ngx.log(ngx.WARN, "jwt verify failed: ", reason, " uri=", uri)
        return response.error(401, reason, 401)
    end

    local payload = jwt_obj.payload

    -- 检查过期（resty-jwt 会自动检查，但显式检查更清晰）
    if payload.exp and payload.exp < ngx.time() then
        return response.error(401, "token expired", 401)
    end

    -- 检查签发者（可选）
    if config.issuer and payload.iss ~= config.issuer then
        return response.error(401, "invalid issuer", 401)
    end

    -- 检查受众（可选）
    if config.audience then
        local aud = payload.aud
        local aud_ok = false
        if type(aud) == "string" then
            aud_ok = (aud == config.audience)
        elseif type(aud) == "table" then
            for _, a in ipairs(aud) do
                if a == config.audience then aud_ok = true; break end
            end
        end
        if not aud_ok then
            return response.error(401, "invalid audience", 401)
        end
    end

    -- 把用户信息存入 ctx，供后续插件和后端使用
    ngx.ctx.user = {
        id    = payload.sub,
        name  = payload.name,
        roles = payload.roles,
        jti   = payload.jti,
    }

    -- 把用户信息注入请求头（传递给后端服务）
    ngx.req.set_header("X-User-Id",    tostring(payload.sub or ""))
    ngx.req.set_header("X-User-Name",  payload.name or "")
    ngx.req.set_header("X-User-Roles", payload.roles and table.concat(payload.roles, ",") or "")

    -- 删除原始 Authorization 头（不透传 Token 到后端，安全起见）
    if config.strip_token then
        ngx.req.clear_header("Authorization")
    end
end

return Plugin

签发 Token（登录接口示例）

鉴权插件只负责验证，Token 的签发通常在独立的登录接口里：

-- handlers/login.lua
local jwt    = require "resty.jwt"
local cjson  = require "cjson"

content_by_lua_block {
    ngx.req.read_body()
    local body = ngx.req.get_body_data()
    local data, err = require("cjson.safe").decode(body or "")

    if not data or not data.username or not data.password then
        ngx.status = 400
        ngx.say('{"code":400,"message":"invalid request"}')
        return
    end

    -- 验证用户名密码（省略具体实现）
    local user, err = db.verify_user(data.username, data.password)
    if not user then
        ngx.status = 401
        ngx.say('{"code":401,"message":"wrong username or password"}')
        return
    end

    local now = ngx.time()
    local jwt_token = jwt:sign(
        os.getenv("JWT_SECRET"),
        {
            header = { typ = "JWT", alg = "HS256" },
            payload = {
                sub   = tostring(user.id),
                name  = user.name,
                roles = user.roles,
                iss   = "my-gateway",
                iat   = now,
                exp   = now + 7200,    -- 2 小时过期
                jti   = ngx.md5(tostring(now) .. user.id),
            }
        }
    )

    ngx.header["Content-Type"] = "application/json"
    ngx.say(cjson.encode({
        code    = 0,
        token   = jwt_token,
        expires = now + 7200,
    }))
}

Token 刷新

JWT 过期后需要重新登录，用户体验差。通常的做法是双 Token 机制：

  • access_token：短期有效（2 小时），用于接口调用
  • refresh_token：长期有效（30 天），用于刷新 access_token
-- 刷新 Token 接口
content_by_lua_block {
    ngx.req.read_body()
    local body = require("cjson.safe").decode(ngx.req.get_body_data() or "")
    local refresh_token = body and body.refresh_token

    if not refresh_token then
        return require("lib.response").error(400, "missing refresh_token")
    end

    -- 验证 refresh_token（用不同的密钥）
    local jwt    = require "resty.jwt"
    local jwt_obj = jwt:verify(os.getenv("REFRESH_SECRET"), refresh_token)

    if not jwt_obj.verified then
        return require("lib.response").error(401, "invalid refresh token", 401)
    end

    -- 检查 refresh_token 是否在黑名单（用户已注销）
    local blacklist = require "lib.token_blacklist"
    if blacklist.is_revoked(jwt_obj.payload.jti) then
        return require("lib.response").error(401, "token revoked", 401)
    end

    -- 重新签发 access_token
    local now = ngx.time()
    local new_token = jwt:sign(os.getenv("JWT_SECRET"), {
        header  = { typ = "JWT", alg = "HS256" },
        payload = {
            sub  = jwt_obj.payload.sub,
            name = jwt_obj.payload.name,
            iat  = now,
            exp  = now + 7200,
            jti  = ngx.md5(tostring(now) .. jwt_obj.payload.sub),
        }
    })

    require("lib.response").success({ token = new_token, expires = now + 7200 })
}

路由级插件配置示例

在路由配置里启用 jwt_auth 插件，并配置白名单：

{
  "id": "user-service",
  "path_prefix": "/api/v1/user",
  "plugins": {
    "jwt_auth": {
      "enabled": true,
      "secret": "your_jwt_secret",
      "issuer": "my-gateway",
      "strip_token": true,
      "whitelist": [
        "/api/v1/user/login",
        "/api/v1/user/register",
        "/api/v1/user/refresh-token"
      ]
    }
  }
}

登录、注册、刷新 Token 这三个接口不需要鉴权，加入白名单即可。

常见问题

时钟偏差问题

JWT 的 exp 是绝对时间戳。如果网关和签发服务的时钟不同步，可能出现 Token 刚签发就"过期"的情况。通常在验证时允许一定的时钟偏差（leeway）：

-- resty-jwt 支持 valid_issuers 和其他选项，时钟偏差需要手动处理
if payload.exp and payload.exp + 30 < ngx.time() then
    -- 允许 30 秒的时钟偏差
    return response.error(401, "token expired", 401)
end

算法混淆攻击

永远不要从 Token 的 Header 里读取算法，然后用读取到的算法验证——这会导致攻击者把算法改成 none 来绕过验证。lua-resty-jwt 默认行为是安全的，不要手动覆盖算法。

小结

JWT 鉴权插件的核心流程：

  1. 从 Authorization: Bearer <token> 或 query string 提取 Token
  2. 白名单路由直接放行
  3. 用密钥验证签名，检查 exp 过期时间
  4. 验证通过后把用户信息存入 ngx.ctx.user，注入请求头传递给后端
  5. 可选：删除原始 Token Header，避免透传到后端
