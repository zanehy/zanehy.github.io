16 | 连接 Redis：实战篇

发表于： 2026-03-24 分类于： OpenResty
字数： 1737 阅读：≈ 4分钟 浏览：

上一篇打好了 Redis 连接的基础。这篇来看实战场景：Pipeline 批量操作、Pub/Sub 订阅，以及两个高频业务需求——分布式 Session 和 Token 黑名单。

Pipeline 批量操作

每次 Redis 命令都是一次网络往返（RTT）。如果需要执行 10 条命令，串行执行就是 10 次 RTT。Pipeline 把多个命令打包成一次网络请求发出去，一次性拿回所有结果，大幅减少网络开销。

local redis = require "resty.redis"
local red = redis:new()
red:set_timeouts(1000, 1000, 1000)
red:connect("127.0.0.1", 6379)

-- 开启 pipeline
red:init_pipeline()

-- 这些命令不会立即发送，先缓存在本地
red:set("key1", "val1")
red:set("key2", "val2")
red:get("key1")
red:get("key2")
red:incr("counter")

-- 一次性发送所有命令，返回结果数组
local results, err = red:commit_pipeline()
if not results then
    ngx.log(ngx.ERR, "pipeline failed: ", err)
    red:set_keepalive(10000, 100)
    return
end

-- results[1] 对应第一条命令的结果，以此类推
ngx.say("set key1: ", results[1])   -- OK
ngx.say("set key2: ", results[2])   -- OK
ngx.say("get key1: ", results[3])   -- val1
ngx.say("get key2: ", results[4])   -- val2
ngx.say("counter: ",  results[5])   -- 1（incr 结果）

red:set_keepalive(10000, 100)

Pipeline 适合写多读少或批量初始化的场景。如果命令之间有依赖（比如先 GET 再根据结果决定 SET），就不能用 Pipeline，要用 Lua Script（EVAL）或者串行执行。

Pipeline 的注意事项 

-- 取消 pipeline（放弃已缓存的命令）
red:cancel_pipeline()

-- pipeline 期间如果连接断开，commit_pipeline 会报错
-- 需要做好重试逻辑

Pub/Sub 订阅

Redis 的发布订阅模型可以用来做简单的消息传递。在 OpenResty 里，通常用在以下场景：

  • 配置变更通知（发布方修改配置后广播，订阅方更新本地缓存）
  • 简单的事件广播

订阅方 

-- 订阅需要在独立的连接上，不能复用业务连接
-- 通常放在 init_worker_by_lua 里用定时器启动

local function start_subscriber(premature)
    if premature then return end

    local redis = require "resty.redis"
    local red = redis:new()
    red:set_timeouts(1000, 60000, 1000)  -- 读超时设长一些

    local ok, err = red:connect("127.0.0.1", 6379)
    if not ok then
        ngx.log(ngx.ERR, "subscriber connect failed: ", err)
        -- 5 秒后重试
        ngx.timer.at(5, start_subscriber)
        return
    end

    -- 订阅频道
    local res, err = red:subscribe("config:update")
    if not res then
        ngx.log(ngx.ERR, "subscribe failed: ", err)
        red:close()
        ngx.timer.at(5, start_subscriber)
        return
    end

    -- 循环读取消息
    while true do
        local msg, err = red:read_reply()
        if not msg then
            ngx.log(ngx.ERR, "read reply failed: ", err)
            red:close()
            -- 断线重连
            ngx.timer.at(1, start_subscriber)
            return
        end

        -- msg = {"message", channel, data}
        if msg[1] == "message" then
            local channel = msg[2]
            local data    = msg[3]
            ngx.log(ngx.INFO, "received on ", channel, ": ", data)

            -- 处理消息，比如更新本地缓存
            local cache = ngx.shared.app_config
            cache:set("config", data)
        end
    end
end

init_worker_by_lua_block {
    ngx.timer.at(0, start_subscriber)
}

发布方 

local red = redis:new()
red:connect("127.0.0.1", 6379)

-- 发布消息
local receivers, err = red:publish("config:update", '{"version":2,"debug":false}')
ngx.log(ngx.INFO, "message delivered to ", receivers, " subscribers")

red:set_keepalive(10000, 100)

实战：分布式 Session 存储

传统 Session 存在服务器内存里，多台服务器之间无法共享。用 Redis 存 Session，所有服务器都能读取，天然支持水平扩展。

-- lib/session.lua
local redis = require "resty.redis"
local cjson = require "cjson"
local M = {}

local SESSION_PREFIX  = "session:"
local SESSION_TTL     = 7200  -- 2 小时

local function get_redis()
    local red = redis:new()
    red:set_timeouts(1000, 1000, 1000)
    local ok, err = red:connect("127.0.0.1", 6379)
    if not ok then return nil, err end
    return red
end

-- 创建 Session
function M.create(user_data)
    local red, err = get_redis()
    if not red then return nil, err end

    -- 生成随机 session_id
    local session_id = ngx.md5(ngx.now() .. ngx.var.remote_addr .. math.random())

    local key = SESSION_PREFIX .. session_id
    local ok, err = red:set(key, cjson.encode(user_data), "EX", SESSION_TTL)

    red:set_keepalive(10000, 100)

    if not ok then return nil, err end
    return session_id
end

-- 读取 Session
function M.get(session_id)
    if not session_id then return nil end

    local red, err = get_redis()
    if not red then return nil end

    local key = SESSION_PREFIX .. session_id
    local val, err = red:get(key)

    red:set_keepalive(10000, 100)

    if not val or val == ngx.null then return nil end
    return cjson.decode(val)
end

-- 刷新过期时间
function M.refresh(session_id)
    local red, err = get_redis()
    if not red then return end
    red:expire(SESSION_PREFIX .. session_id, SESSION_TTL)
    red:set_keepalive(10000, 100)
end

-- 销毁 Session（退出登录）
function M.destroy(session_id)
    local red, err = get_redis()
    if not red then return end
    red:del(SESSION_PREFIX .. session_id)
    red:set_keepalive(10000, 100)
end

return M

在请求处理中使用：

local session = require "lib.session"

-- 登录时创建 Session
local session_id, err = session.create({
    user_id   = 123,
    user_name = "alice",
    role      = "admin",
})
ngx.header["Set-Cookie"] = "sid=" .. session_id .. "; HttpOnly; Path=/"

-- 后续请求验证 Session
local cookies = ngx.req.get_headers()["cookie"] or ""
local sid = cookies:match("sid=([^;]+)")

local user = session.get(sid)
if not user then
    return ngx.redirect("/login")
end

session.refresh(sid)  -- 滑动过期时间
ngx.ctx.user = user

实战：Token 黑名单

JWT Token 本身是无状态的——一旦签发，在过期之前一直有效。但有些场景需要提前让 Token 失效（用户主动退出、账号被封禁、密码修改后旧 Token 作废）。

用 Redis 黑名单是最常见的解决方案：把已失效的 Token 的 JTI（JWT ID）或哈希存入 Redis，验证时额外检查一下。

-- lib/token_blacklist.lua
local redis = require "resty.redis"
local M = {}

local BLACKLIST_PREFIX = "token:blacklist:"

local function get_redis()
    local red = redis:new()
    red:set_timeouts(1000, 1000, 1000)
    local ok, err = red:connect("127.0.0.1", 6379)
    if not ok then return nil, err end
    return red
end

-- 加入黑名单（ttl 设为 Token 剩余有效期，到期自动清理）
function M.revoke(jti, ttl)
    local red, err = get_redis()
    if not red then return false, err end

    local key = BLACKLIST_PREFIX .. jti
    local ok, err = red:set(key, "1", "EX", ttl)

    red:set_keepalive(10000, 100)
    return ok, err
end

-- 检查是否在黑名单
function M.is_revoked(jti)
    local red, err = get_redis()
    if not red then
        -- Redis 不可用时的策略：拒绝（安全优先）或放行（可用性优先）
        -- 这里选择拒绝
        return true
    end

    local val, err = red:get(BLACKLIST_PREFIX .. jti)
    red:set_keepalive(10000, 100)

    return val ~= nil and val ~= ngx.null
end

return M

在鉴权中间件里使用：

local jwt = require "resty.jwt"
local blacklist = require "lib.token_blacklist"

local auth_header = ngx.req.get_headers()["authorization"] or ""
local token = auth_header:match("Bearer (.+)")

if not token then
    return ngx.exit(401)
end

-- 验证签名和过期时间
local jwt_obj = jwt:verify("your_secret", token)
if not jwt_obj.verified then
    return ngx.exit(401)
end

-- 检查黑名单
local jti = jwt_obj.payload.jti
if blacklist.is_revoked(jti) then
    ngx.log(ngx.WARN, "revoked token used: ", jti)
    return ngx.exit(401)
end

ngx.ctx.user_id = jwt_obj.payload.sub

用户退出时：

local exp = jwt_obj.payload.exp
local ttl = exp - ngx.time()
if ttl > 0 then
    blacklist.revoke(jti, ttl)
end

小结

Redis 进阶用法的几个核心：

  • Pipeline：批量命令打包发送，减少 RTT，适合无依赖的批量操作
  • Pub/Sub：用独立连接订阅，注意断线重连和 premature 处理
  • 分布式 Session：用 Redis 存储，支持水平扩展，注意滑动过期
  • Token 黑名单：存 JTI，TTL 设为 Token 剩余有效期，Redis 不可用时按安全策略处理
