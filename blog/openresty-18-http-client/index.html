18 | HTTP 客户端

发表于： 2026-03-26 分类于： OpenResty
字数： 1766 阅读：≈ 4分钟 浏览：

在 API 网关的场景里，OpenResty 不只是接收请求，很多时候还需要主动调用其他服务——验证 Token、查询用户信息、发送 Webhook 通知。这些都需要 HTTP 客户端能力。

lua-resty-http

lua-resty-http 是目前 OpenResty 生态里最成熟的 HTTP 客户端库，由 pintsized 维护。它基于 cosocket，完全非阻塞。

安装（如果未内置）：

opm get ledgetech/lua-resty-http

基本用法

简单 GET 请求 

local httpc = require("resty.http").new()

-- request_uri 是最简洁的接口，适合简单请求
local res, err = httpc:request_uri("https://api.example.com/users/123", {
    method  = "GET",
    headers = {
        ["Authorization"] = "Bearer " .. token,
        ["Accept"]        = "application/json",
    },
    ssl_verify = false,  -- 开发环境可关闭证书验证，生产环境建议开启
})

if not res then
    ngx.log(ngx.ERR, "request failed: ", err)
    return ngx.exit(500)
end

ngx.log(ngx.INFO, "status: ", res.status)
ngx.say(res.body)

POST 请求（JSON Body） 

local httpc = require("resty.http").new()
local cjson = require "cjson"

local payload = cjson.encode({
    user_id = 123,
    action  = "login",
    ip      = ngx.var.remote_addr,
})

local res, err = httpc:request_uri("https://audit.internal/events", {
    method  = "POST",
    body    = payload,
    headers = {
        ["Content-Type"]   = "application/json",
        ["Content-Length"] = #payload,
    },
})

if not res then
    ngx.log(ngx.ERR, "audit request failed: ", err)
    -- 审计失败不影响主流程，这里不退出
    return
end

if res.status ~= 200 and res.status ~= 201 then
    ngx.log(ngx.WARN, "audit returned: ", res.status, " ", res.body)
end

POST 表单 

local res, err = httpc:request_uri("https://api.example.com/token", {
    method = "POST",
    body   = ngx.encode_args({
        grant_type    = "client_credentials",
        client_id     = "my_app",
        client_secret = "secret",
    }),
    headers = {
        ["Content-Type"] = "application/x-www-form-urlencoded",
    },
})

超时配置

超时设置是 HTTP 客户端使用中最容易被忽略的细节，但它直接影响服务的稳定性。

local httpc = require("resty.http").new()

-- 分别设置连接超时、发送超时、读取超时（毫秒）
httpc:set_timeouts(
    2000,   -- connect_timeout：建立连接的超时时间
    5000,   -- send_timeout：发送请求的超时时间
    10000   -- read_timeout：等待响应的超时时间
)

三个超时的含义：

  • connect_timeout：从发起连接到 TCP 握手完成的时间。如果目标服务不可达，这个超时决定了你等多久才放弃
  • send_timeout：发送请求数据（特别是大 body）的超时时间
  • read_timeout：从发送完请求到收到完整响应的超时时间。对于慢接口（比如需要处理复杂计算），这个值要设大一点

生产环境建议：所有外部 HTTP 调用都设置超时，且不能设太大（避免慢接口拖垮整个 worker）。

连接池

和 Redis 一样，用完 HTTP 连接后要归还连接池，避免频繁建立 TCP 连接的开销。

local httpc = require("resty.http").new()

local res, err = httpc:request_uri("http://backend-service/api", {
    method = "GET",
    keepalive = true,           -- 启用 keepalive（默认就是 true）
    keepalive_timeout = 60000,  -- 连接在池中保持 60 秒
    keepalive_pool    = 100,    -- 连接池大小
})

-- request_uri 会自动处理连接归还，不需要手动调用

如果使用底层的 connect + request 接口，需要手动管理：

local httpc = require("resty.http").new()
httpc:set_timeouts(2000, 5000, 10000)

-- 连接到目标服务器
local ok, err = httpc:connect({
    scheme = "http",
    host   = "backend-service",
    port   = 8080,
})
if not ok then
    ngx.log(ngx.ERR, "connect failed: ", err)
    return
end

-- 发请求
local res, err = httpc:request({
    method  = "GET",
    path    = "/api/users",
    headers = { ["Host"] = "backend-service" },
})

-- 读取响应（必须读完，否则连接不能复用）
local body, err = res:read_body()

-- 归还连接池
httpc:set_keepalive(60000, 100)

实战：调用第三方鉴权接口

很多企业有统一的鉴权服务，API 网关需要把每个请求的 Token 发到鉴权服务验证，拿到用户信息后再转发到业务服务。

-- lib/auth_client.lua
local httpc_module = require "resty.http"
local cjson = require "cjson"
local M = {}

local AUTH_SERVICE = "http://auth.internal"
local TIMEOUT_MS   = 2000

function M.verify(token)
    local httpc = httpc_module.new()
    httpc:set_timeouts(TIMEOUT_MS, TIMEOUT_MS, TIMEOUT_MS)

    local res, err = httpc:request_uri(AUTH_SERVICE .. "/verify", {
        method  = "POST",
        body    = cjson.encode({ token = token }),
        headers = {
            ["Content-Type"] = "application/json",
        },
        keepalive_timeout = 30000,
        keepalive_pool    = 50,
    })

    if not res then
        ngx.log(ngx.ERR, "auth service unavailable: ", err)
        -- 鉴权服务不可用时的策略：拒绝请求
        return nil, "auth service unavailable"
    end

    if res.status == 401 then
        return nil, "invalid token"
    end

    if res.status ~= 200 then
        ngx.log(ngx.ERR, "auth service error: ", res.status)
        return nil, "auth service error"
    end

    local data, err = cjson.decode(res.body)
    if not data then
        return nil, "invalid auth response"
    end

    return data  -- {user_id, user_name, roles, ...}
end

return M

在 access_by_lua 阶段使用：

access_by_lua_block {
    local auth = require "lib.auth_client"

    local auth_header = ngx.req.get_headers()["authorization"] or ""
    local token = auth_header:match("Bearer (.+)")

    if not token then
        ngx.status = 401
        ngx.say('{"code":401,"message":"missing token"}')
        return ngx.exit(ngx.OK)
    end

    local user, err = auth.verify(token)
    if not user then
        ngx.status = 401
        ngx.say('{"code":401,"message":"' .. err .. '"}')
        return ngx.exit(ngx.OK)
    end

    -- 把用户信息存入上下文，供后续阶段使用
    ngx.ctx.user = user
    -- 注入到请求头，传递给后端服务
    ngx.req.set_header("X-User-Id",   tostring(user.user_id))
    ngx.req.set_header("X-User-Name", user.user_name)
}

实战：Webhook 通知

Webhook 是事件驱动集成的常见方式——某个事件发生时，主动 POST 通知到对方的 URL。

在 OpenResty 里，Webhook 通知通常要异步发送（不能阻塞主请求）：

-- lib/webhook.lua
local httpc_module = require "resty.http"
local cjson = require "cjson"
local M = {}

-- 异步发送 Webhook（不阻塞当前请求）
function M.send_async(url, payload)
    -- 用定时器异步执行
    ngx.timer.at(0, function(premature, url, payload)
        if premature then return end

        local httpc = httpc_module.new()
        httpc:set_timeouts(3000, 5000, 10000)

        local body = cjson.encode(payload)
        local res, err = httpc:request_uri(url, {
            method  = "POST",
            body    = body,
            headers = {
                ["Content-Type"]  = "application/json",
                ["X-Webhook-Time"] = tostring(ngx.time()),
            },
            keepalive_timeout = 10000,
            keepalive_pool    = 20,
        })

        if not res then
            ngx.log(ngx.ERR, "webhook failed to ", url, ": ", err)
            return
        end

        if res.status >= 400 then
            ngx.log(ngx.WARN, "webhook ", url, " returned: ", res.status)
        end
    end, url, payload)
end

return M

在业务逻辑里调用：

local webhook = require "lib.webhook"

-- 订单完成后通知
webhook.send_async("https://partner.example.com/hooks/order", {
    event    = "order.completed",
    order_id = 12345,
    amount   = 99.9,
    time     = ngx.time(),
})

-- 立即返回，不等待 Webhook 结果
ngx.say('{"code":0,"message":"ok"}')

处理大响应体

默认情况下 request_uri 会把整个响应体读入内存。如果响应体很大，需要流式处理：

local httpc = require("resty.http").new()
local ok, err = httpc:connect({scheme="http", host="files.internal", port=80})

local res, err = httpc:request({ method="GET", path="/large-file.csv" })
if not res then return end

-- 流式读取，每次读 4KB
local reader = res.body_reader
while true do
    local chunk, err = reader(4096)
    if err then
        ngx.log(ngx.ERR, "read error: ", err)
        break
    end
    if not chunk then break end  -- 读完了
    ngx.print(chunk)
    ngx.flush(true)
end

httpc:set_keepalive(60000, 50)

小结

lua-resty-http 的使用要点：

  • request_uri 适合简单请求，自动处理连接归还；底层 connect + request 更灵活但需手动管理连接
  • 务必设置超时，三个超时分别对应不同阶段
  • 连接池通过 keepalive_timeout 和 keepalive_pool 配置
  • 鉴权调用放在 access_by_lua 阶段，结果存 ngx.ctx 传递
  • Webhook 等不影响主流程的通知，用 ngx.timer.at(0, ...) 异步发送
